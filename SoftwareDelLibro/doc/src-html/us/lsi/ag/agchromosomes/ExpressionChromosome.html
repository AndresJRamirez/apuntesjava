035        /**
036         * Número de bits para codificar cada uno de los valores enteros resultantes
037         */
038        public static Integer numeroDeBits = 5;
039
040        /**
041         * Problema a resolver
042         */
043        public static ProblemaAGExpression<?,?> problem;
044
045        /**
046         * Longitud de la cabeza de cada uno de los genes
047         */
048        public static Integer headLeng; 
049
050        private static Integer tailLeng;
051        
052        /**
053         * Número de Genes del cromosoma
054         */
055        public static Integer numGenes;
056        
057        private static Integer numItemsPorGen;

059        private static Integer numItems;

069        /**
070         * @pos Inicializa los parámetros de la clase
071         * @param problema El problema a resolver
072         */
073        public static <T> void iniValues(ProblemaAG problema) {
074                ExpressionChromosome.problem = (ProblemaAGExpression<?, ?>) problema;
075                int maxArity = ExpressionChromosome.problem.getOperators().stream()
076                                .mapToInt(x -> x.getArity()).max().getAsInt();
077                ExpressionChromosome.tailLeng = ExpressionChromosome.headLeng * (maxArity - 1) + 1;
078                ExpressionChromosome.numItemsPorGen = ExpressionChromosome.headLeng + ExpressionChromosome.tailLeng;            
079                ExpressionChromosome.numItems = ExpressionChromosome.numItemsPorGen*ExpressionChromosome.numGenes + 
080                                ExpressionChromosome.problem.getNumConstants(); 
081                ExpressionChromosome.constantsIndex = ExpressionChromosome.numItemsPorGen*ExpressionChromosome.numGenes;
082                ExpressionChromosome.DIMENSION = ExpressionChromosome.numeroDeBits*ExpressionChromosome.numItems;
083        }
084        
085        /**
086         * @param i Un indice a 
087         * @return
088         */
089        private Integer getMax(int i) {         
090                Integer r = null;
091                if (i>=ExpressionChromosome.constantsIndex) {
092                        r = ExpressionChromosome.problem.getMaxValueConstant();
093                } else {
094                        i = i%ExpressionChromosome.numItemsPorGen;
095                        if (i < ExpressionChromosome.headLeng) {
096                                r = this.getNumVariables()+this.getNumConstants()+this.getNumOperators();
097                        } else if (i < ExpressionChromosome.headLeng+ExpressionChromosome.tailLeng) {
098                                r = this.getNumVariables()+this.getNumConstants();
099                        } 
100                }
101                return r;
102        }

104        private static Integer pow = Math2.pow(2., numeroDeBits).intValue();
105
106        private List<VariableExp<T>> variables;
107        private List<ConstantExp<T>> constants;
108        private Exp<T> expression;
109        private List<Integer> items;
110        private double ft;
111        private ProblemaAGExpression<?,T> prb;
112
113        @SuppressWarnings("unchecked")
114        public ExpressionChromosome(Integer[] representation)
115                        throws InvalidRepresentationException {
116                super(representation);
117                this.prb = (ProblemaAGExpression<?, T>) ExpressionChromosome.problem;           
118                this.items = this.dcd();
119                this.variables = this.getNumVariables(this.getNumVariables());
120            this.constants = this.getNumConstants(this.getNumConstants());
121            int operatorIndex = this.getNumVariables()+this.getNumConstants();;
122            List<Exp<T>> ls = Lists.newArrayList();
123            ls.addAll(this.variables);
124            ls.addAll(this.constants);
125            ls.addAll(this.prb.getOperators());
126            List<Exp<T>> expressions = Lists.newArrayList();
127            int index1 = 0;
128                for (int i = 0; i < ExpressionChromosome.numGenes; i++) {
129                        int index2 = index1 +  ExpressionChromosome.numItemsPorGen;
130                        List<List<Exp<T>>> levels = Exp.levels(this.items.subList(index1, index2), operatorIndex, ls);
131                        Exp<T> expression = levels.get(0).get(0);
132                        index1 = index1+ ExpressionChromosome.numItemsPorGen;
133                        expressions.add(expression);
134                }
135                NaryExp<T> nExp = this.prb.getNaryExp().clone();
136                nExp.setOps(expressions);
137                this.expression = nExp;
138                this.ft = this.calculateFt();
139        }

142        @SuppressWarnings("unchecked")
143        public ExpressionChromosome(List<Integer> representation)
144                        throws InvalidRepresentationException {
145                super(representation);
146                this.prb = (ProblemaAGExpression<?, T>) ExpressionChromosome.problem;           
147                this.items = this.dcd();
148                this.variables = this.getNumVariables(this.getNumVariables());
149            this.constants = this.getNumConstants(this.getNumConstants());
150            int operatorIndex = this.getNumVariables()+this.getNumConstants();;
151            List<Exp<T>> ls = Lists.newArrayList();
152            ls.addAll(this.variables);
153            ls.addAll(this.constants);
154            ls.addAll(this.prb.getOperators());
155            List<Exp<T>> expressions = Lists.newArrayList();
156            int index1 = 0;
157                for (int i = 0; i < ExpressionChromosome.numGenes; i++) {
158                        int index2 = index1 +  ExpressionChromosome.numItemsPorGen;
159                        List<List<Exp<T>>> levels = Exp.levels(this.items.subList(index1, index2), operatorIndex, ls);
160                        Exp<T> expression = levels.get(0).get(0);
161                        index1 = index1+ ExpressionChromosome.numItemsPorGen;
162                        expressions.add(expression);
163                }
164                NaryExp<T> nExp = this.prb.getNaryExp().clone();
165                nExp.setOps(expressions);
166                this.expression = nExp;
167                this.ft = this.calculateFt();
168        }

170        @Override
171        public AbstractListChromosome<Integer> newFixedLengthChromosome(List<Integer> ls) {
172                return new ExpressionChromosome<T>(ls);
173        }
174        
175        @Override
176        public Exp<T> decode() {
177                return this.getExp();
178        }
179
180        private List<Integer> dcd() {
181                List<Integer> ls = super.getRepresentation();
182                List<Integer> r = new ArrayList<Integer>();
183                int index1 = 0;
184                for (int i = 0; i < ExpressionChromosome.numItems; i++) {
185                        int index2 = index1 + numeroDeBits;
186                        Integer e = Lists2.decode(ls.subList(index1, index2));
187                        Integer d = Math2.escala(e, pow, this.getMax(i));
188                        r.add(d);
189                        index1 = index1 + numeroDeBits;
190                }
191                return r;
192        }

258        /**
259         * Identificadores de las variables a usar
260         */
261        public static List<String> nombres = Arrays.asList("x","y","z"+"t","u","v"+"w");
262                        
263        private List<VariableExp<T>> getNumVariables(int num){
264                List<VariableExp<T>> ls = Lists.newArrayList();
265                for(int i=0; i<num;i++){
266                        ls.add(Exp.<T>createVariable(nombres.get(i)));
267                }
268                return ls;
269        }
270        
271        private List<ConstantExp<T>> getNumConstants(int num){
272                List<ConstantExp<T>> ls = Lists.newArrayList();
273                for(int i=0; i<num;i++){
274                        ls.add(Exp.<T>createConstant(this.prb.convert(this.items.get(ExpressionChromosome.constantsIndex+i))));
275                }
276                return ls;
277        }
