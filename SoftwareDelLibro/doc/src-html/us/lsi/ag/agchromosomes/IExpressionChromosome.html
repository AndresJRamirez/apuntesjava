001package us.lsi.ag.agchromosomes;
002
003import us.lsi.ag.ProblemaAGExpression;
004import us.lsi.common.ConstantExp;
005import us.lsi.common.Exp;
006import us.lsi.common.VariableExp;
007
008/**
009 * @author Miguel Toro
010 *
011 * @param <T> El tipo del resultado de la expresión
012 * 
013 * <p> Representa un cromosoma cuyo valor decodificado es
014 * una expresión formada por un conjunto de operadores dados, un conjunto de variables y un conjunto de constantes </p>
015 */
016public interface IExpressionChromosome<T> extends IChromosome<Exp<T>> {
017        
018        /**
019         * @return El problema que se va a resolver
020         */
021        ProblemaAGExpression<?,T> getProblem();
022        
023        /**
024         * @return Numero de operadores
025         */
026        Integer getNumOperators();
027        
028        /**
029         * @return Número de variables
030         */
031        Integer getNumVariables();
032        
033        /**
034         * @return Número de constantes
035         */
036        Integer getNumConstants();
037        
038        /**
039         * @param i Un indice en el rango <code> 0..getNumOperators()-1 </code>
040         * @return El operador de índice i
041         */
042        Exp<T> getOperator(int i);
043        
044        /**
045         * @param i Un índice en el rango <code> 0..getNumVariables()-1 </code>
046         * @return La vraible de índice i
047         */
048        VariableExp<T> getVariable(int i);
049        
050        /**
051         * @param i Un índice en el rango <code> 0..getNumConstants()-1 </code>
052         * @return la constante de índice i
053         */
054        ConstantExp<T> getConstant(int i);
055        
056        /**
057         * @return la expresión asociada al cromosoma
058         */
059        Exp<T> getExp();        
060
061}
