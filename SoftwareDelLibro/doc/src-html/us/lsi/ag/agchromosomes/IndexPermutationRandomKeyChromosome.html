001package us.lsi.ag.agchromosomes;
002
003import java.util.List;
004
005import org.apache.commons.math3.genetics.AbstractListChromosome;
006import org.apache.commons.math3.genetics.Chromosome;
007import org.apache.commons.math3.genetics.InvalidRepresentationException;
008import org.apache.commons.math3.genetics.RandomKey;
009
010import us.lsi.ag.ProblemaAG;
011import us.lsi.ag.ProblemaAGIndex;
012
013import com.google.common.base.Preconditions;
014
015/**
016 * @author Miguel Toro
017 * 
018 * <p> Una implementación del tipo IndexChromosome. Toma como información la definición de un problema que implementa el interfaz ProblemaAGIndex.
019 * A partir de esa información construye una secuencia normal. Asumimos que el número de objetos es n y el tamaño de la secuencia normal r. 
020 *  
021 * <p> La lista decodificada está formada por una lista de  tamaño r, cuyos valores son 
022 * índices en el rango [0,n-1], y cada índice i se  repite un número de veces dado por la multiplicidad máxima del objeto i
023 * definida en el problema. </p>
024 * 
025 * <p> La implementación usa un cromosoma de clave aleatoria de tamaño r.
026 * Es un cromosoma adecuado para codificar problemas de permutaciones </p>
027 *
028 */
029public class IndexPermutationRandomKeyChromosome extends RandomKey<Integer> implements IndexChromosome {
030
031        public static List<Integer> normalSequence = null;
032        public static ProblemaAGIndex<?> problema;
033        
034        /**
035         * Dimensión del cromosoma
036         */
037        
038        public static int DIMENSION;
039        
040        public static void iniValues(ProblemaAG problema){
041                IndexPermutationRandomKeyChromosome.problema = (ProblemaAGIndex<?>) problema; 
042                IndexPermutationRandomKeyChromosome.normalSequence = IndexPermutationRandomKeyChromosome.problema.getNormalSequence();
043                IndexPermutationRandomKeyChromosome.DIMENSION = IndexPermutationRandomKeyChromosome.normalSequence.size();
044        }
045
046        
047        public IndexPermutationRandomKeyChromosome(List<Double> representation)
048                        throws InvalidRepresentationException {
049                super(representation);
050                this.ft = this.calculateFt();
051        }
052
053        public IndexPermutationRandomKeyChromosome(Double[] representation)
054                        throws InvalidRepresentationException {
055                super(representation);
056                this.ft = this.calculateFt();
057        }
058
059        @Override
060        public AbstractListChromosome<Double> newFixedLengthChromosome(List<Double> ls) {
061                return new IndexPermutationRandomKeyChromosome(ls);
062                
063        }
064
065        @Override
066        public List<Integer> decode() {
067                Preconditions.checkArgument(IndexPermutationRandomKeyChromosome.normalSequence!=null);
068                return this.decode(IndexPermutationRandomKeyChromosome.normalSequence);
069        }
070        
071        
072        public static IndexPermutationRandomKeyChromosome getInitialChromosome() {
073                List<Double> ls = RandomKey.randomPermutation(IndexPermutationRandomKeyChromosome.DIMENSION);
074                return new IndexPermutationRandomKeyChromosome(ls);
075        }
076
077        
078        @Override
079        public double fitness() {
080                return ft;
081        }
082        
083        private double ft;
084        
085        private double calculateFt(){
086                return IndexPermutationRandomKeyChromosome.problema.fitnessFunction(this);
087        }
088
089        @Override
090        public ProblemaAGIndex<?> getProblem() {
091                return IndexPermutationRandomKeyChromosome.problema;
092        }
093
094        @Override
095        public Integer getObjectsNumber() {
096                return IndexPermutationRandomKeyChromosome.problema.getObjectsNumber();
097        }
098
099        @Override
100        public Integer getMax(int i) {
101                return IndexPermutationRandomKeyChromosome.problema.getMax(i);
102        }
103        
104        @Override
105        public Chromosome asChromosome() {
106                return this;
107        }
108        
109}
