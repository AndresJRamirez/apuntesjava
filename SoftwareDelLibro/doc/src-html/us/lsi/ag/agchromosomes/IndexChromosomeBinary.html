001package us.lsi.ag.agchromosomes;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.apache.commons.math3.genetics.AbstractListChromosome;
007import org.apache.commons.math3.genetics.BinaryChromosome;
008import org.apache.commons.math3.genetics.Chromosome;
009import org.apache.commons.math3.genetics.InvalidRepresentationException;
010
011import us.lsi.ag.ProblemaAG;
012import us.lsi.ag.ProblemaAGIndex;
013import us.lsi.common.Lists2;
014import us.lsi.math.Math2;
015
016/**
017 * @author Miguel Toro
018 * 
019 * 
020 * <p> Una implementación del tipo Cromosoma&lt;Integer&gt;. Toma como información la definición de un problema que implementa el interfaz ProblemaAGIndex.
021 * Asumimos que el número de objetos es n. Usa una  secuencia normal implícita de tamaño n de la forma (0,1,2, ..., n-1). </p>
022 * 
023 * <p> La lista decodificada está formada por una lista de  tamaño n cuyos elementos para cada i son 
024 *  valores en en rango [0,m(i)], siendo m(i) la multiplicidad máxima para i. </p>
025 * 
026 * <p> La implementación usa un cromosoma binario del tamaño n*nbits. 
027 * Siendo nbits el número de bits usados para representar cada uno de los enteros. </p>
028 * 
029 * <p> Es un cromosoma adecuado para codificar problemas de subconjuntos de multiconjuntos</p>
030 *
031 */
032public class IndexChromosomeBinary extends BinaryChromosome implements IndexChromosome {
033        
034        public static Integer numeroDeBits = 5;
035        
036        public static ProblemaAGIndex<?> problema;
037        
038        /**
039         * Dimensión del cromosoma
040         */
041        
042        public static int DIMENSION;
043        
044        public static void iniValues(ProblemaAG problema){
045                IndexChromosomeBinary.problema = (ProblemaAGIndex<?>) problema; 
046                IndexChromosomeBinary.DIMENSION = IndexChromosomeBinary.numeroDeBits*IndexChromosomeBinary.problema.getObjectsNumber();
047        }
048        
049        private static Integer pow = Math2.pow(2., numeroDeBits).intValue();
050        
051        public IndexChromosomeBinary(Integer[] representation) throws InvalidRepresentationException {
052                super(representation);
053                this.ft = this.calculateFt();
054        }
055
056        public IndexChromosomeBinary(List<Integer> representation) throws InvalidRepresentationException {
057                super(representation);
058                this.ft = this.calculateFt();
059        }
060
061        @Override
062        public AbstractListChromosome<Integer> newFixedLengthChromosome(List<Integer> ls) {
063                return new IndexChromosomeBinary(ls);
064        }
065        
066        public List<Integer> decode() {
067                List<Integer> ls = super.getRepresentation();
068                List<Integer> r = new ArrayList<Integer>();
069                int index1 = 0;
070                for(int i = 0; i < getObjectsNumber(); i++){                    
071                        int index2 = index1+numeroDeBits;
072                        Integer e = Lists2.decode(ls.subList(index1, index2));
073                        Integer d = Math2.escalaIncluded(e, pow, getMax(i));
074                        r.add(d);
075                        index1 = index1+numeroDeBits;;
076                }
077                return r;
078        }
079
080        public List<Integer> getRepresentation(){
081                return super.getRepresentation();
082        }
083        
084        public static IndexChromosomeBinary getInitialChromosome() {
085                List<Integer> ls = BinaryChromosome.randomBinaryRepresentation(IndexChromosomeBinary.DIMENSION);
086                return new IndexChromosomeBinary(ls);
087        }
088
089        @Override
090        public double fitness() {
091                return ft;
092        }
093        
094        private double ft;
095        
096        private double calculateFt(){
097                return IndexChromosomeBinary.problema.fitnessFunction(this);
098        }
099
100        @Override
101        public Integer getObjectsNumber() {
102                return IndexChromosomeBinary.problema.getObjectsNumber();
103        }
104
105        @Override
106        public Integer getMax(int i) {
107                return IndexChromosomeBinary.problema.getMax(i);
108        }
109
110        @Override
111        public ProblemaAGIndex<?> getProblem() {
112                return IndexChromosomeBinary.problema;
113        }
114
115        @Override
116        public Chromosome asChromosome() {
117                return this;
118        }
119        
120        
121        
122}
