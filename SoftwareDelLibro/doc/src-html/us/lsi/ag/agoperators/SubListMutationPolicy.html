001package us.lsi.ag.agoperators;
002
003import org.apache.commons.math3.exception.MathIllegalArgumentException;
004import org.apache.commons.math3.genetics.BinaryChromosome;
005import org.apache.commons.math3.genetics.BinaryMutation;
006import org.apache.commons.math3.genetics.Chromosome;
007import org.apache.commons.math3.genetics.MutationPolicy;
008import org.apache.commons.math3.genetics.RandomKey;
009import org.apache.commons.math3.genetics.RandomKeyMutation;
010
011import us.lsi.ag.agchromosomes.IndexChromosomePermutationSubList;
012
013import com.google.common.base.Preconditions;
014
015/**
016 * <p> Un operador de mutación adecaudo para cromosomas de tipo mixto y que reutiliza los operadores mutación proporcionados
017 * en <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/package-summary.html" 
018 * target="_blank"> Apache Genetics </a>
019 * </p>
020 *   
021 * @author Miguel Toro
022 *
023 */
024
025public class SubListMutationPolicy implements MutationPolicy {
026
027        protected static MutationPolicy binaryOperator = new BinaryMutation();
028        protected static MutationPolicy randomKeyOperator  = new RandomKeyMutation();
029        
030        public SubListMutationPolicy() {
031                super();
032        }
033        
034        @Override
035        public Chromosome mutate(Chromosome chr0) throws MathIllegalArgumentException {
036                if (!(chr0 instanceof IndexChromosomePermutationSubList))
037                        throw new IllegalArgumentException();;
038                IndexChromosomePermutationSubList c0 = (IndexChromosomePermutationSubList) chr0;                
039                BinaryChromosome binary = c0.getBinary();
040                RandomKey<Integer> randomKey = c0.getRandomKey();
041                Preconditions.checkArgument(binary != null);            
042                Preconditions.checkArgument(randomKey != null); 
043                Chromosome c1 = binaryOperator.mutate(binary);
044                Preconditions.checkArgument(c1 instanceof BinaryChromosome);
045                Chromosome c2 = randomKeyOperator.mutate(randomKey);            
046                Preconditions.checkArgument(c2 instanceof RandomKey);
047                return new IndexChromosomePermutationSubList(c1,c2);
048        }       
049        
050}
