001package us.lsi.ag.mochila;
002
003import java.util.List;
004
005import org.apache.commons.math3.genetics.AbstractListChromosome;
006import org.apache.commons.math3.genetics.BinaryChromosome;
007import org.apache.commons.math3.genetics.InvalidRepresentationException;
008
009
010
011import us.lsi.pd.mochila.ProblemaMochila;
012
013public class CromosomaMochila extends BinaryChromosome {
014
015        public CromosomaMochila(Integer[] representation)
016                        throws InvalidRepresentationException {
017                super(representation);
018                
019        }
020
021        public CromosomaMochila(List<Integer> representation)
022                        throws InvalidRepresentationException {
023                super(representation);
024                
025        }
026
027        @Override
028        public double fitness() {
029                double r;
030                Integer peso = 0;
031                Integer valor = 0;
032                for (int i = 0; i < super.getRepresentation().size(); i++) {
033                        int w = super.getRepresentation().get(i);
034                        peso = peso + w*ProblemaMochila.getPesoObjeto(ProblemaMochilaAG.indicesDeObjetos.get(i));
035                        valor = valor + w*ProblemaMochila.getValorObjeto(ProblemaMochilaAG.indicesDeObjetos.get(i));
036                }
037                double dif = ProblemaMochila.capacidadInicial - peso;
038                if(dif >= 0.){
039                        r = valor;
040                } else {
041                        r = valor + 10000*(dif);
042                }
043                return r;
044        }
045
046        @Override
047        public AbstractListChromosome<Integer> newFixedLengthChromosome(List<Integer> ls) {
048                return new CromosomaMochila(ls);
049        }
050
051        @Override
052        public List<Integer> getRepresentation(){
053                return super.getRepresentation();
054        }
055
056}
