001package us.lsi.ag.mochila;
002
003import java.util.List;
004
005import us.lsi.ag.Cromosoma;
006import us.lsi.ag.ProblemaAGBag;
007import us.lsi.pd.mochila.ObjetoMochila;
008import us.lsi.pd.mochila.ProblemaMochila;
009import us.lsi.pd.mochila.SolucionMochila;
010
011public class ProblemaMochilaAG implements ProblemaAGBag<SolucionMochila> {
012
013        public ProblemaMochilaAG(String fichero) {
014                ProblemaMochila.leeObjetosDisponibles(fichero);
015        }       
016
017        @Override
018        public SolucionMochila getSolucion(Cromosoma<Integer> chromosome) {
019                SolucionMochila s = SolucionMochila.create();
020                for (Integer e: chromosome.decode()) {
021                        s =     s.add(ProblemaMochila.getObjeto(e),1);
022                }
023                return s;
024        }
025        
026        
027
028        @Override
029        public Double fitnessFunction(List<Integer> ls) {
030                double r;
031                Integer peso = 0;
032                Integer valor = 0;
033        for (Integer e: ls) {
034                        peso = peso + ProblemaMochila.getPesoObjeto(e);
035                        valor = valor + ProblemaMochila.getValorObjeto(e);
036                }
037                double dif = ProblemaMochila.capacidadInicial - peso;
038                if(dif >= 0.){
039                        r = valor;
040                } else {
041                        r = valor - 10000*(dif*dif);
042                }
043                return r;
044        }
045
046        
047        public List<ObjetoMochila> getObjetos() {
048                return ProblemaMochila.getObjetosDisponibles();
049        }
050        
051        @Override
052        public Integer getMultiplicidadMaxima(int index){
053                return ProblemaMochila.getObjetosDisponibles().get(index).getNumMaxDeUnidades();
054        }
055
056        @Override
057        public Integer getNumeroDeObjetos() {
058                return this.getObjetos().size();
059        }
060        
061        
062        
063}
