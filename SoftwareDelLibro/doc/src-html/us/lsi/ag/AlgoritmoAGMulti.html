001package us.lsi.ag;
002
003import org.apache.commons.math3.genetics.BinaryMutation;
004import org.apache.commons.math3.genetics.Chromosome;
005
006import com.google.common.collect.Lists;
007
008public class AlgoritmoAGMulti extends AlgoritmoAG<Integer> {
009        
010        public AlgoritmoAGMulti(ProblemaAGBag<?> problema) {
011                super();        
012                super.mutationPolicy = new BinaryMutation();    
013                BagMultiChromosome.numeroDeObjetos = problema.getNumeroDeObjetos();
014                AlgoritmoAG.DIMENSION = BagMultiChromosome.numeroDeBits*BagMultiChromosome.numeroDeObjetos;
015                BagMultiChromosome.multiplicidades = Lists.newArrayList();
016                for(int i=0; i< BagMultiChromosome.numeroDeObjetos;i++){
017                        BagMultiChromosome.multiplicidades.add(problema.getMultiplicidadMaxima(i));
018                }
019        }
020        
021        @Override
022        public Chromosome getInitialChromosome() {
023                return BagMultiChromosome.getInitialChromosome();
024        }
025}
