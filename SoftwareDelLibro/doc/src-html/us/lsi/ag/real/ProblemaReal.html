001package us.lsi.ag.real;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import us.lsi.ag.Cromosoma;
007import us.lsi.ag.ProblemaAGReal;
008import us.lsi.common.Par;
009
010public class ProblemaReal implements ProblemaAGReal<List<Double>> {
011
012        public ProblemaReal(){
013                
014        }
015
016        @Override
017        public List<Double> getSolucion(Cromosoma<Double> chromosome) {
018                return chromosome.decode();
019        }
020
021        @Override
022        public Integer getNumeroDeVariables() {
023                return 2;
024        }
025
026        @Override
027        public List<Par<Double, Double>> getLimites() {
028                List<Par<Double,Double>> ls = new ArrayList<>();
029                ls.add(Par.create(0., 1.));
030                ls.add(Par.create(0., 15.));
031                return ls;
032        }
033
034        @Override
035        public Double fitnessFunction(List<Double> ls) {
036                Double a = ls.get(0);
037                Double b = ls.get(1);
038//              return (100 * Math.pow(Math.pow(a,2) - b,2) + Math.pow(1 - a,2))-constraint(a,b);
039                return -(a*a+b*b)+constraint(a,b);
040        }
041            
042            private double constraint(Double a, Double b){
043                boolean r = true;
044//              r = r && (a*b + a - b + 1.5) <=0;
045//              r = r && (10 - a*b <=0);
046                return r?0:10000000;
047            }
048                
049}
