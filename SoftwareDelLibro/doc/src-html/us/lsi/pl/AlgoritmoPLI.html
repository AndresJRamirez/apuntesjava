001package us.lsi.pl;
002
003import java.util.List;
004import java.util.function.Function;
005
006import com.google.common.collect.Lists;
007
008import us.lsi.algoritmos.AbstractAlgoritmo;
009import us.lsi.common.StringExtensions2;
010import lpsolve.LpSolve;
011import lpsolve.LpSolveException;
012
013public class AlgoritmoPLI  extends AbstractAlgoritmo implements SolutionPL{
014
015        public static AlgoritmoPLI create(IProblemaPL p, String fichero) {
016                p.toStringConstraints(fichero);
017                return new AlgoritmoPLI(fichero);
018        }
019        
020        public static AlgoritmoPLI create(String fichero) {
021                return new AlgoritmoPLI(fichero);
022        }
023        
024        public static AlgoritmoPLI create() {
025                return new AlgoritmoPLI("intermedio.txt");
026        }
027        
028        private double[] solutionPoint;
029        private double solutionValue;
030        private List<String> names;
031        
032        private String fichero;
033        
034        private LpSolve solver;
035
036        private AlgoritmoPLI(String fichero) {
037                super();
038                this.fichero = fichero;
039        }
040        
041        /**
042         * @return El coste total del objetivo
043         */
044        public double getObjetivo(){
045                return solutionValue;
046        }
047
048        /**
049         * @return El punto solución
050         */
051        public double[] getSolucion() {
052                return solutionPoint;
053        }
054
055        /**
056         * @return La solución de la variable i
057         */
058        public double getSolucion(int i) {
059                return solutionPoint[i];
060        }
061                
062        public List<String> getNames() {
063                return names;
064        }
065
066        public String getName(int i) {
067                return names.get(i);
068        }
069        
070        public int getNumVar() {
071                return solutionPoint.length;
072        }
073        
074        
075        public String getFichero() {
076                return fichero;
077        }
078
079        public void setConstraints(String r){
080                StringExtensions2.toFile(r,this.getFichero());
081        }
082        
083        public void ejecuta(){
084                
085                
086                try {
087                        
088                        solver = LpSolve.readLp(this.fichero, 1, "Problema");
089
090                        solver.solve();
091                        solutionPoint = solver.getPtrVariables();
092                        solutionValue = solver.getObjective();
093                        names = Lists.newArrayList();
094                        for (int j = 1; j <= solutionPoint.length; j++) {
095                                names.add(solver.getOrigcolName(j));
096                        }
097                        solver.deleteLp();
098                        
099                } catch (LpSolveException e) {
100                        throw new IllegalStateException("Se ha producido una excepción en LpSolve");
101                }
102
103        }
104
105        public static String getFactor(Integer num, String symbol){
106                return num+"*"+symbol;
107        }
108
109        public static String getSumVariable(String name, int i1, int i2){
110                String r = "";
111                for(int i =i1;i<i2;i++){
112                        if (i!=i1) r = r+"+";
113                        r = r +AlgoritmoPLI.getVariable(name,i);
114                }       
115                return r;
116        }
117        
118        public static String getSumFactor(Function<Integer,Double> f, String name, int i1, int i2){
119                String r = "";
120                for(int i =i1;i<i2;i++){
121                        if (i!=i1) r = r+"+";
122                        r = r +f.apply(i)+"+"+AlgoritmoPLI.getVariable(name,i);
123                }       
124                return r;
125        }
126        
127        public static String getVariable(String name, int i, int j, int k){
128                return name+i+"_"+j+"_"+k;
129        }
130
131        public static String getVariable(String name, int i, int j){
132                return name+i+"_"+j;
133        }
134
135        public static String getVariable(String name, int index){
136                return name+index;
137        }
138        
139        public static String intVariables(String name, int i1, int i2) {
140                String r = "";
141                r = r +"int ";
142                for(int i =i1;i<i2;i++){
143                        if (i!=0) r = r+",";
144                        r = r +AlgoritmoPLI.getVariable("x",i);
145                }
146                r = r+";\n\n";
147                return r;
148        }
