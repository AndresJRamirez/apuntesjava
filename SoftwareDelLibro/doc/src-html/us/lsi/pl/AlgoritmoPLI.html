001package us.lsi.pl;
002
003import lpsolve.LpSolve;
004import lpsolve.LpSolveException;
005
006public class AlgoritmoPLI implements SolutionPL {
007
008        public static AlgoritmoPLI create(IProblemaPL p, String fichero) {
009                p.toStringConstraints(fichero);
010                return new AlgoritmoPLI(fichero);
011        }
012        
013        public static AlgoritmoPLI create(String fichero) {
014                return new AlgoritmoPLI(fichero);
015        }
016        
017        private double[] solutionPoint;
018        private double solutionValue;
019        
020        private String fichero;
021        
022        private LpSolve solver;
023
024        private AlgoritmoPLI(String fichero) {
025                super();
026                this.fichero = fichero;
027        }
028        
029        /**
030         * @return El coste total del objetivo
031         */
032        public double getObjetivo(){
033                return solutionValue;
034        }
035
036        /**
037         * @return El punto solución
038         */
039        public double[] getSolucion() {
040                return solutionPoint;
041        }
042
043        
044        
045        
046        public void ejecuta(){
047                try {
048                        solver = LpSolve.readLp(this.fichero, 1, "Problema");
049
050                        solver.solve();
051                        solutionPoint = solver.getPtrVariables();
052                        solutionValue = solver.getObjective();
053
054                        // delete the problem and free memory
055                        solver.deleteLp();
056                } catch (LpSolveException e) {
057                        e.printStackTrace();
058                }
059
060        }
061        
062        
063        
064}
