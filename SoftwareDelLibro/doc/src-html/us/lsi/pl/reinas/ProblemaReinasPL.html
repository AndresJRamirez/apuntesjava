001package us.lsi.pl.reinas;
002
003import org.apache.commons.math3.optim.linear.Relationship;
004
005import us.lsi.algoritmos.Algoritmos;
006import us.lsi.pl.AlgoritmoPLI;
007import us.lsi.pl.ProblemaPL;
008import us.lsi.pl.ProblemaPL.TipoDeOptimizacion;
009import us.lsi.pl.ProblemaPL.TipoDeVariable;
010
011public class ProblemaReinasPL {
012
013        public static int numeroDeReinas = 8;
014        
015        public static int getPos(int i, int j){
016                return i+j*numeroDeReinas;
017        }
018        
019        public static ProblemaPL getProblema(){
020                int n = numeroDeReinas;
021                int nV = numeroDeReinas*numeroDeReinas;
022                ProblemaPL problema = ProblemaPL.create(nV, TipoDeOptimizacion.Max);
023                double[] fo = new double[nV];
024                for (int i = 0; i < nV; i++) {
025                        fo[i] = 1.;
026                }
027                problema.setObjectiveFunction(fo, 0);
028                
029                for (int i = 0; i < n; i++) {
030                        double[] columnas = new double[nV];
031                        for (int j = 0; j < n; j++) {
032                                int p = getPos(i,j);
033                                columnas[p] = 1;
034                        }
035                        problema.addConstraint(columnas, Relationship.EQ, 1);
036                }
037                
038                for (int j = 0; j < n; j++) {
039                        double[] filas = new double[nV];
040                        for (int i = 0; i < n; i++) {
041                                int p = getPos(i,j);
042                                filas[p] = 1;
043                        }
044                        problema.addConstraint(filas, Relationship.EQ, 1);
045                }
046                
047                for (int k = -n+1; k < n; k++) {
048                        double[] dp = new double[nV];
049                        for (int i = -n+1; i < n; i++) {
050                                int j = i+k;
051                                if (i>=0 && i<n && j>=0 && j<n) {
052                                        int p = getPos(i, j);
053                                        dp[p] = 1;
054                                }
055                        }
056                        problema.addConstraint(dp, Relationship.LEQ, 1);
057                }
058                
059                for (int k = 0; k <= 2*n-2; k++) {
060                        double[] ds = new double[nV];
061                        for (int i = 0; i <= 2*n-2; i++) {
062                                int j = -i+k;
063                                if (i>=0 && i<n && j>=0 && j<n) {
064                                        int p = getPos(i, j);
065                                        ds[p] = 1;
066                                }
067                        }
068                        problema.addConstraint(ds, Relationship.LEQ, 1);
069                }
070                
071                
072                problema.setTipoDeTodasLasVariables(TipoDeVariable.Binaria);
073                return problema;
074        }
075        
076        
077        public static void main(String[] args) {
078                numeroDeReinas = 100;
079                ProblemaPL p = ProblemaReinasPL.getProblema();
080                p.toStringConstraints("ficheros\\reinas.txt");
081                AlgoritmoPLI a = Algoritmos.createPLI("ficheros\\reinas.txt");
082                a.ejecuta();            
083                System.out.println("________");
084                System.out.println(Math.round(a.getObjetivo()));
085                System.out.println("________");
086                System.out.println("________");
087                String s ="";
088                for (int pos = 0; pos < p.getNumOfVariables(); pos++) {
089                        int i = pos%numeroDeReinas;
090                        double x = a.getSolucion()[pos];
091                        if (Math.round(x)==1) {
092                                s= s+ " X ";
093                        }else{
094                                s= s+ "   ";
095                        }
096                        if(i==numeroDeReinas-1){
097                                s = s+"\n";
098                        }
099                }
100                System.out.println(s);
101                System.out.println("________");
102        }
103}
