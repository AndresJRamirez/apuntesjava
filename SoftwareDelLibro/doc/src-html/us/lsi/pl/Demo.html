001package us.lsi.pl;
002
003import lpsolve.*;
004
005public class Demo {
006
007  public static void main(String[] args) {
008    try {
009      // Create a problem with 4 variables and 0 constraints
010      LpSolve solver = LpSolve.makeLp(0, 4);
011
012      // add constraints
013      double r1[] = {3, 2, 2, 1};
014      double r2[] = {0, 3, 3, 1};
015      double r3[] = {2, 3, -2, 3};
016//      solver.strAddConstraint("3 2 2 1", LpSolve.LE, 4);
017//      solver.strAddConstraint("0 4 3 1", LpSolve.LE, 3);
018
019      solver.addConstraint(r1, LpSolve.LE, 4);
020      solver.addConstraint(r2, LpSolve.GE, 3);
021      // set objective function
022 //     solver.strSetObjFn("2 3 -2 3");
023      solver.setObjFn(r3);
024      
025      solver.setMaxim();;;
026      // solve the problem
027      solver.solve();
028System.out.println("______________________");
029      solver.printConstraints(0);;
030      solver.printConstraints(1);
031      solver.printTableau();
032      solver.printSolution(0);
033System.out.println("______________________");     
034      // print solution
035      System.out.println("Value of objective function: " + solver.getObjective());
036      double[] var = solver.getPtrVariables();
037      for (int i = 0; i < var.length; i++) {
038        System.out.println("Value of var[" + i + "] = " + var[i]);
039      }
040
041      // delete the problem and free memory
042      solver.deleteLp();
043    }
044    catch (LpSolveException e) {
045       e.printStackTrace();
046    }
047  }
048
049}
