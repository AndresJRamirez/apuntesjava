001package us.lsi.pl;
002
003import java.util.Collection;
004
005import org.apache.commons.math3.optim.PointValuePair;
006import org.apache.commons.math3.optim.linear.LinearConstraint;
007import org.apache.commons.math3.optim.linear.LinearConstraintSet;
008import org.apache.commons.math3.optim.linear.LinearObjectiveFunction;
009import org.apache.commons.math3.optim.linear.NonNegativeConstraint;
010import org.apache.commons.math3.optim.linear.SimplexSolver;
011import org.apache.commons.math3.optim.nonlinear.scalar.GoalType;
012
013import us.lsi.flowgraph.NoSeEncuentraSolucion;
014import us.lsi.pl.ProblemaPL.TipoDeOptimizacion;
015
016public class AlgoritmoPL implements SolutionPL {
017
018        public static AlgoritmoPL create(IProblemaPL p){
019                return new AlgoritmoPL(p.getObjectiveFunction(), p.getConstraints(), p.getTipo());
020        }
021
022        public static AlgoritmoPL create(IProblemaPL p, String fichero){
023                p.toStringConstraints(fichero);
024                return new AlgoritmoPL(p.getObjectiveFunction(), p.getConstraints(), p.getTipo());
025        }
026        
027        private double[] solutionPoint;
028        private double solutionValue;
029        
030        private TipoDeOptimizacion tipo;
031        private LinearObjectiveFunction objectiveFunction;
032        private Collection<LinearConstraint> collection;
033        
034        
035        private AlgoritmoPL(LinearObjectiveFunction objectiveFunction, Collection<LinearConstraint> collection, 
036                        TipoDeOptimizacion tipo) {
037                super();
038                this.tipo = tipo;
039                this.objectiveFunction = objectiveFunction;
040                this.collection = collection;
041        }
042        
043        private AlgoritmoPL(LinearObjectiveFunction objectiveFunction, Collection<LinearConstraint> collection) {
044                super();
045                this.tipo = TipoDeOptimizacion.Min;
046                this.objectiveFunction = objectiveFunction;
047                this.collection = collection;
048        }
049
050        /**
051         * @return El coste total del objetivo
052         */
053        public double getObjetivo(){
054                return solutionValue;
055        }
056
057        /**
058         * @return El punto solución
059         */
060        public double[] getSolucion() {
061                return solutionPoint;
062        }
063
064        /**
065         * Calcula la solución
066         * @exception NoSeEncuentraSolucion Si no existe solución o el algoritmo no puede encontrarla
067         */
068        public void ejecuta(){
069                SimplexSolver ss = new SimplexSolver(); 
070                LinearConstraintSet lcs = new LinearConstraintSet(collection);
071                PointValuePair p = null;
072                try {
073                        p = ss.optimize(objectiveFunction, lcs,tipo.equals(TipoDeOptimizacion.Min)?GoalType.MINIMIZE:GoalType.MAXIMIZE,
074                                        new NonNegativeConstraint(true));
075                } catch (Exception e) {
076                        throw new NoSeEncuentraSolucion();
077                }               
078                solutionPoint = p.getPoint();
079                solutionValue = p.getValue();
080        }
081
082        public TipoDeOptimizacion getTipo() {
083                return tipo;
084        }
085
086        public LinearObjectiveFunction getObjectiveFunction() {
087                return objectiveFunction;
088        }
089
090
091        public Collection<LinearConstraint> getCollection() {
092                return collection;
093        }
094        
095}
