001package us.lsi.pl;
002
003import java.util.Collection;
004
005import org.apache.commons.math3.optim.PointValuePair;
006import org.apache.commons.math3.optim.linear.LinearConstraint;
007import org.apache.commons.math3.optim.linear.LinearConstraintSet;
008import org.apache.commons.math3.optim.linear.LinearObjectiveFunction;
009import org.apache.commons.math3.optim.linear.NonNegativeConstraint;
010import org.apache.commons.math3.optim.linear.SimplexSolver;
011import org.apache.commons.math3.optim.nonlinear.scalar.GoalType;
012
013import us.lsi.flowgraph.NoSeEncuentraSolucion;
014import us.lsi.pl.ProblemaPL.TipoDeOptimizacion;
015/**
016 * @author Miguel Toro
017 * 
018 * <p> Un algoritmo de Programación Lineal. El algoritmo parte de una definición del problema recogida en un tipo 
019 *  {@link us.lsi.pl.ProblemaPL ProblemaPL}  y calcula la solución. El algoritmo solo admite variables reales. </p>
020 * 
021 * 
022 * <p> La implementación reutiliza el algoritmo SimplexSolver de Apache. </p>
023 */
024public class AlgoritmoPL implements SolutionPL {
025
026        public static AlgoritmoPL create(IProblemaPL p){
027                return new AlgoritmoPL(p.getObjectiveFunction(), p.getConstraints(), p.getTipo());
028        }
029
030        public static AlgoritmoPL create(IProblemaPL p, String fichero){
031                p.toStringConstraints(fichero);
032                return new AlgoritmoPL(p.getObjectiveFunction(), p.getConstraints(), p.getTipo());
033        }
034        
035        private double[] solutionPoint;
036        private double solutionValue;
037        
038        private TipoDeOptimizacion tipo;
039        private LinearObjectiveFunction objectiveFunction;
040        private Collection<LinearConstraint> collection;
041        
042        
043        private AlgoritmoPL(LinearObjectiveFunction objectiveFunction, Collection<LinearConstraint> collection, 
044                        TipoDeOptimizacion tipo) {
045                super();
046                this.tipo = tipo;
047                this.objectiveFunction = objectiveFunction;
048                this.collection = collection;
049        }
050        
051        private AlgoritmoPL(LinearObjectiveFunction objectiveFunction, Collection<LinearConstraint> collection) {
052                super();
053                this.tipo = TipoDeOptimizacion.Min;
054                this.objectiveFunction = objectiveFunction;
055                this.collection = collection;
056        }
057
058        /**
059         * @return El coste total del objetivo
060         */
061        public double getObjetivo(){
062                return solutionValue;
063        }
064
065        /**
066         * @return El punto solución
067         */
068        public double[] getSolucion() {
069                return solutionPoint;
070        }
071
072        /**
073         * Calcula la solución
074         * @exception NoSeEncuentraSolucion Si no existe solución o el algoritmo no puede encontrarla
075         */
076        public void ejecuta(){
077                SimplexSolver ss = new SimplexSolver(); 
078                LinearConstraintSet lcs = new LinearConstraintSet(collection);
079                PointValuePair p = null;
080                try {
081                        p = ss.optimize(objectiveFunction, lcs,tipo.equals(TipoDeOptimizacion.Min)?GoalType.MINIMIZE:GoalType.MAXIMIZE,
082                                        new NonNegativeConstraint(true));
083                } catch (Exception e) {
084                        throw new NoSeEncuentraSolucion();
085                }               
086                solutionPoint = p.getPoint();
087                solutionValue = p.getValue();
088        }
089
090        public TipoDeOptimizacion getTipo() {
091                return tipo;
092        }
093
094        public LinearObjectiveFunction getObjectiveFunction() {
095                return objectiveFunction;
096        }
097
098
099        public Collection<LinearConstraint> getCollection() {
100                return collection;
101        }
102        
103        
104}
