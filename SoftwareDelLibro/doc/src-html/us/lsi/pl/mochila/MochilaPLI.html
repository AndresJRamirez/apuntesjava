001package us.lsi.pl.mochila;
002
003import us.lsi.pd.mochila.ProblemaMochila;
004import us.lsi.pl.AlgoritmoPLI;
005
006public class MochilaPLI {
007
008        
009        public static String getConstraints(){
010                ProblemaMochila.leeObjetosDisponibles(AlgoritmoPLI.getRaiz()+"objetosMochila.txt");
011                ProblemaMochila.capacidadInicial = 78;
012                int num = ProblemaMochila.getObjetosDisponibles().size();
013                String r = "";
014                r = r+"max: ";
015                for(int i =0;i<num;i++){
016                        if (i!=0) r = r+"+";
017                        r = r +AlgoritmoPLI.getFactor(ProblemaMochila.getValorObjeto(i),AlgoritmoPLI.getVariable("x",i));
018                }               
019                r = r+";\n\n";
020                for(int i =0;i<num;i++){
021                        if (i!=0) r = r+"+";
022                        r = r +AlgoritmoPLI.getFactor(ProblemaMochila.getPesoObjeto(i),AlgoritmoPLI.getVariable("x",i));
023                }
024                r = r +"<=";
025                r = r +ProblemaMochila.capacidadInicial;
026                r = r+";\n\n";
027                for(int i =0;i<num;i++){
028                        r = r +AlgoritmoPLI.getVariable("x",i)+"<="+ProblemaMochila.getNumMaxDeUnidades(i)+";\n";
029                }
030                r = r +"int ";
031                for(int i =0;i<num;i++){
032                        if (i!=0) r = r+",";
033                        r = r +AlgoritmoPLI.getVariable("x",i);
034                }
035                r = r+";\n";            
036                return r;
037        }
038
039}
