001package us.lsi.concurrent.filosofos;
002
003
004import com.google.common.base.Preconditions;
005
006
007public class TenedoresSync {
008        
009        private static int nt = FilosofosSync.numeroDeFilosofos;
010        private static Tenedor[] tenedores;
011        
012        public static void init() {
013                tenedores = new Tenedor[nt];
014                
015                for(int i =0;i <nt; i++){
016                        tenedores[i] = new Tenedor(i);
017                }
018        }
019        
020        public static int getNumeroDeTenedores() {
021                return nt;
022        }
023        
024        public static Tenedor[] getTenedores() {
025                return tenedores;
026        }
027        
028        public static Tenedor getTenedor(int i) {
029                Preconditions.checkElementIndex(i, nt);
030                return tenedores[i];
031        }
032
033}
