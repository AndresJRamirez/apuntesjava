001package us.lsi.concurrent.filosofos;
002
003
004public class Tenedor extends AbstractModelObject {
005        
006        public enum Estado {Libre, Ocupado};
007        
008        private Estado estado = Estado.Libre;
009        private int i;
010        
011        public Tenedor(int i) {
012                this.i = i;
013        }
014
015        public void setEstado(Estado estado) {
016                Estado oldEstado = this.estado;
017                boolean oldDisponible = (this.estado == Estado.Libre);
018                this.estado = estado;
019                firePropertyChange("estado", oldEstado, estado);
020                firePropertyChange("disponible", oldDisponible, isDisponible());
021                System.out.println("Tenedor  "+ i+","+isDisponible());
022        }
023
024        public Estado getEstado() {
025                return estado;
026        }
027        
028        public boolean isDisponible(){  
029                return estado == Estado.Libre;
030        }
031
032        public int getNumero() {
033                return i;
034        }
035        
036        
037}
