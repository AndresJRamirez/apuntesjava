001package us.lsi.concurrent.algoritmos;
002import java.util.*;
003
004public class SequentialPuzzleSolver<P, M> {
005        private final Puzzle<P, M> puzzle;
006        private final Set<P> seen = new HashSet<P>();
007
008        public SequentialPuzzleSolver(Puzzle<P, M> puzzle) {
009                this.puzzle = puzzle;
010        }
011
012        public List<M> solve() {
013                P pos = puzzle.initialPosition();
014                return search(new PuzzleNode<P, M>(pos, null, null));
015        }
016
017        private List<M> search(PuzzleNode<P, M> node) {
018                if (!seen.contains(node.pos)) {
019                        seen.add(node.pos);
020                        if (puzzle.isGoal(node.pos))
021                                return node.asMoveList();
022                        for (M move : puzzle.legalMoves(node.pos)) {
023                                P pos = puzzle.move(node.pos, move);
024                                PuzzleNode<P, M> child = new PuzzleNode<P, M>(pos, move, node);
025                                List<M> result = search(child);
026                                if (result != null)
027                                        return result;
028                        }
029                }
030                return null;
031        }
032
033}
