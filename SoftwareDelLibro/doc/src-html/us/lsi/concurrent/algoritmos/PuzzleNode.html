001package us.lsi.concurrent.algoritmos;
002
003import java.util.*;
004
005import net.jcip.annotations.*;
006
007@Immutable
008public class PuzzleNode<P, M> {
009        final P pos;
010        final M move;
011        final PuzzleNode<P, M> prev;
012
013        public PuzzleNode(P pos, M move, PuzzleNode<P, M> prev) {
014                this.pos = pos;
015                this.move = move;
016                this.prev = prev;
017        }
018
019        List<M> asMoveList() {
020                List<M> solution = new LinkedList<M>();
021                for (PuzzleNode<P, M> n = this; n.move != null; n = n.prev)
022                        solution.add(0, n.move);
023                return solution;
024        }
025
026}
