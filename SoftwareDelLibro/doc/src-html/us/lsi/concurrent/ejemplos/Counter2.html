001package us.lsi.concurrent.ejemplos;
002
003import java.util.concurrent.atomic.*;
004
005import net.jcip.annotations.ThreadSafe;
006
007@ThreadSafe
008public class Counter2 {
009        
010        private AtomicLong value = new AtomicLong();
011        
012        public long getValue() {
013              return value.longValue();
014        }
015
016        public long increment() {
017              if (value.longValue() == Long.MAX_VALUE)
018                    throw new IllegalStateException("counter overflow");
019              return value.incrementAndGet();
020        }
021}
