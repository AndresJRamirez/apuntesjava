001package us.lsi.concurrent.ejemplos;
002
003public class LaunderThrowable {
004         /**
005     * Coerce an unchecked Throwable to a RuntimeException
006     * If the Throwable is an Error, throw it; if it is a
007     * RuntimeException return it, otherwise throw IllegalStateException
008     * @param t La excepción de origen
009     * @return  La excepción convertida
010     */
011    public static RuntimeException launderThrowable(Throwable t) {
012        if (t instanceof RuntimeException)
013            return (RuntimeException) t;
014        else if (t instanceof Error)
015            throw (Error) t;
016        else
017            throw new IllegalStateException("Not unchecked", t);
018    }
019
020}
