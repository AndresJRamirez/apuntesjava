001package us.lsi.db.example;
002
003import us.lsi.db.TableDefinition;
004
005
006public class Person implements TableDefinition{
007
008        public Person(){};
009        
010        //Campos en la base de datos
011        private int id ;
012        private String fname ;
013        private String lname ;
014        
015        public int getId() {
016                return id;
017        }
018        public void setId(int id) {
019                this.id = id;
020        }
021        public String getFname() {
022                return fname;
023        }
024        public void setFname(String fname) {
025                this.fname = fname;
026        }
027        public String getLname() {
028                return lname;
029        }
030        public void setLname(String lname) {
031                this.lname = lname;
032        }
033        
034        
035        @Override
036        public String toString() {
037                return "Persons [id=" + id + ", fname=" + fname + ", lname=" + lname
038                                + "]";
039        }
040
041}
