001package us.lsi.db.example;
002
003import java.util.List;
004
005import javax.sql.DataSource;
006
007import us.lsi.db.Mapper;
008
009public class PersonAddressViewDAO {
010        private DataSource ds = null;
011        
012        @SuppressWarnings("unused")
013        private PersonAddressViewDAO(){}
014        
015        public PersonAddressViewDAO(DataSource ds){
016                this.ds = ds;
017        }
018        
019        public List<PersonAddressView> getAllRows() throws Exception {
020                
021                String sql= "SELECT " +
022                                        "p.lname as lastName, p.fname as firstName, a.zip as zipCode " +
023                                        "FROM   " +
024                                        "test.persons p INNER JOIN test.addresses a " +
025                                        "ON " +
026                                        "p.id = a.personid";
027                
028                List<PersonAddressView> personAddressViewRows = new Mapper<PersonAddressView>().fetchData(ds, PersonAddressView.class, sql);
029                
030                return personAddressViewRows;
031
032        }
033}
