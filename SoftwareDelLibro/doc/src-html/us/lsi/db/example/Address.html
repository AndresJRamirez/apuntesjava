001package us.lsi.db.example;
002
003import us.lsi.db.TableDefinition;
004
005public class Address implements TableDefinition{
006        private int id;
007        private int personid;
008        private String streetno;
009        private String streetname;
010        private String city;
011        private String state;
012        private String zip;
013
014        public int getId() {
015                return id;
016        }
017
018        public void setId(int id) {
019                this.id = id;
020        }
021
022        public int getPersonid() {
023                return personid;
024        }
025
026        public void setPersonid(int personid) {
027                this.personid = personid;
028        }
029
030        public String getStreetno() {
031                return streetno;
032        }
033
034        public void setStreetno(String streetno) {
035                this.streetno = streetno;
036        }
037
038        public String getStreetname() {
039                return streetname;
040        }
041
042        public void setStreetname(String streetname) {
043                this.streetname = streetname;
044        }
045
046        public String getCity() {
047                return city;
048        }
049
050        public void setCity(String city) {
051                this.city = city;
052        }
053
054        public String getState() {
055                return state;
056        }
057
058        public void setState(String state) {
059                this.state = state;
060        }
061
062        public String getZip() {
063                return zip;
064        }
065
066        public void setZip(String zip) {
067                this.zip = zip;
068        }
069
070        @Override
071        public String toString() {
072                return "Address [id=" + id + ", personid=" + personid + ", streetno="
073                                + streetno + ", streetname=" + streetname + ", city=" + city
074                                + ", state=" + state + ", zip=" + zip + "]";
075        }
076
077}
