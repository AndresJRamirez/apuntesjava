001package us.lsi.voraz;
002
003/**
004*<p> Tipo mutable del estado de un problema que se quiere resolver por una técnica Voraz. 
005 * Cada estado está asociado un problema, del conjunto de problemas considerado, más el camino, secuencia de alternativas, para llegar a él desde el problema inicial. </p>
006 * 
007 * <p>La documentación puede encontarse en el: <a href="../../../document/Tema15.pdf" target="_blank">Tema15</a></p>
008*
009*
010* @param <E> Tipo del Estado 
011* @param <S> Tipo de la solución del problema
012* @param <A> Tipo de las alternativas
013*/
014public interface EstadoVZ<E extends EstadoVZ<E,S,A>,S,A> {      
015        /**
016         * @param a - Alternativa elegida
017         * @return Siguiente estado si se escoge la alternativa a
018         */
019        E next(A a);    
020        /**
021         * @return La alternativa escogida en el estado actual. 
022         */
023        A getAlternativa();
024        /**
025         * @return Solución del problema calculada a partir del estado actual en el estado que cumple la condición de parada o null si no hay solución
026         */
027        S getSolucion();
028        /**
029         * La condición de parada establece cuando se parará el algoritmo. Esta condición se puede establecer a partir
030         * de la propiedades fijadas para el algoritmo: número de soluciones encontradas, número de iteraciones, etc.
031         * 
032         * @return Si se cumple la condición del parada del algortimo.
033         */
034        boolean condicionDeParada();
035}
