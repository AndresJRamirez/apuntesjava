001package us.lsi.dyv;
002
003import java.util.List;
004
005
006/**
007 * @author Miguel Toro
008 *
009 * @param <S> El tipo de la solución
010 * @param <E> El tipo de las soluciones intermedias
011 */
012public interface ProblemaDyV<S,E> {     
013        /**
014         * @constraint El tamaño de un problema debe ser mayor que el de cada uno de sus subproblemas
015         * @return El tamaño de un problema
016         */
017        int size();
018        /**
019         * @return Si el problema es un caso base
020         */
021        boolean esCasoBase();
022        /**
023         * @return La solución del caso base
024         */
025        E getSolucionCasoBase();
026        /**
027         * @constraint Si todos los valores en ls son null la solución devuelta es null
028         * @pre ls!=null
029         * @param ls - Soluciones de los subproblemas alcanzados tras tomar las diferentes alterantivas
030         * @return La solución del problema
031         */
032        E combina(List<E> ls);
033        /**
034         * @pre 0 &lt; = i &lt; getNumeroSubProblemas(a)
035         * @param i - Número del subproblema
036         * @return Subproblema
037         */
038        ProblemaDyV<S,E> getSubProblema(int i);
039        /**
040         * @return Número de subproblemas
041         */
042        int getNumeroDeSubProblemas();  
043        /**
044         * @param e - Solución parcial
045         * @return Solución construida a partir de la solución parcial
046         */
047        S getSolucion(E e);
048}
