001package us.lsi.function;
002
003import java.util.Comparator;
004
005public class Predicates {
006        
007        public static <T extends Object & Comparable<? super T>> boolean esMayor(T e1, T e2){
008                return e1.compareTo(e2) > 0;
009        }
010        public static <T extends Object & Comparable<? super T>> boolean esMayorOIgual(T e1, T e2){
011                return e1.compareTo(e2) >= 0;
012        }
013        public static <T extends Object & Comparable<? super T>> boolean esMenor(T e1, T e2){
014                return e1.compareTo(e2) < 0;
015        }
016        public static <T extends Object & Comparable<? super T>> boolean esMenorOigual(T e1, T e2){
017                return e1.compareTo(e2) <= 0;
018        }
019        public static <T> boolean esMayor(Comparator<? super T> cmp, T e1, T e2){
020                return cmp.compare(e1,e2) > 0;
021        }
022        public static <T> boolean esMayorOIgual(Comparator<? super T> cmp, T e1, T e2){
023                return cmp.compare(e1,e2) >= 0;
024        }
025        public static <T> boolean esMenor(Comparator<? super T> cmp, T e1, T e2){
026                return cmp.compare(e1,e2) < 0;
027        }
028        public static <T> boolean esMenorOigual(Comparator<? super T> cmp, T e1, T e2){
029                return cmp.compare(e1,e2) <= 0;
030        }
031        public static <T> boolean esIgual(Comparator<? super T> cmp, T e1, T e2){
032                return cmp.compare(e1,e2) == 0;
033        }
034        public static <T> boolean estaEnAbierto(Comparator<? super T> cmp, T e, T e1, T e2){
035                return cmp.compare(e,e1) > 0 && cmp.compare(e,e2) < 0;
036        }
037        public static <T extends Object & Comparable<? super T>> boolean estaEnAbierto(T e, T e1, T e2){
038                return e.compareTo(e1) > 0 && e.compareTo(e2) < 0;
039        }
040        public static <T> boolean estaEnCerrado(Comparator<? super T> cmp, T e, T e1, T e2){
041                return cmp.compare(e,e1) >= 0 && cmp.compare(e,e2) <= 0;
042        }
043        public static <T extends Object & Comparable<? super T>> boolean estaEnCerrado(T e, T e1, T e2){
044                return e.compareTo(e1) >= 0 && e.compareTo(e2) <= 0;
045        }
046}
