001package us.lsi.bt.sudoku;
002
003import java.util.List;
004import java.util.stream.IntStream;
005
006import com.google.common.collect.Lists;
007
008import us.lsi.algoritmos.ProblemaSA;
009import us.lsi.common.ParInteger;
010
011import us.lsi.stream.Stream2;
012
013public class ProblemaSudokuSAAp2 extends ProblemaSudoku implements ProblemaSA<EstadoSudokuSAAp2, CuadroSudoku, ParInteger> {
014
015        public ProblemaSudokuSAAp2(String nf) {
016                super(nf);
017        }
018        
019        @Override
020        public EstadoSudokuSAAp2 getEstadoInicial() {
021                CuadroSudoku cuadro = CuadroSudoku.create();
022
023                List<List<Integer>> valoresLibresEnSubCuadro = Lists.newArrayList();
024
025                for (int sc = 0; sc < ProblemaSudoku.numeroDeFilas; sc++) {
026                        List<Integer> ls = Stream2.create(
027                                        IntStream.rangeClosed(1, ProblemaSudoku.numeroDeFilas)
028                                                        .boxed()).toList();
029                        ls.removeAll(cuadro.getValoresOcupadosEnSubCuadro().get(sc));
030                        valoresLibresEnSubCuadro.add(ls);
031                }
032                for (int sc = 0; sc < ProblemaSudoku.numeroDeFilas; sc++) {
033                        int a = 0;
034                        for (Integer p : ProblemaSudoku.posicionesLibresEnSubCuadro.get(sc)) {
035                                cuadro.setInfo(p, valoresLibresEnSubCuadro.get(sc).get(a));
036                                a++;
037                        }
038                }
039                return EstadoSudokuSAAp2.create(cuadro);
040        }
041
042}
