001package us.lsi.bt.sudoku;
002
003import java.util.Collections;
004import java.util.Comparator;
005import java.util.Map;
006
007import com.google.common.collect.Maps;
008
009import us.lsi.bt.ProblemaBT;
010
011public class ProblemaSudokuBT extends ProblemaSudoku implements ProblemaBT<CuadroSudoku, Integer> {
012        
013        
014        private static Map<Integer, Integer> numOcupadosPorPosicion;
015        public static Comparator<Integer> cmp = (Integer x, Integer y) -> 
016                        ProblemaSudokuBT.numOcupadosPorPosicion
017                        .get(x).compareTo(ProblemaSudokuBT.numOcupadosPorPosicion.get(y));
018
019        public ProblemaSudokuBT(String nf) {
020                super(nf);
021                ProblemaSudokuBT.numOcupadosPorPosicion = ProblemaSudokuBT.calculaNumOcupados();
022                Collections.sort(ProblemaSudoku.posicionesLibres,ProblemaSudokuBT.cmp.reversed());
023        }
024        
025        @Override
026        public EstadoSudokuBT getEstadoInicial() {
027                return EstadoSudokuBT.create();
028        }
029
030        private static Map<Integer,Integer> calculaNumOcupados(){
031                Map<Integer,Integer> map = Maps.newHashMap();
032                CuadroSudoku cuadro = CuadroSudoku.create();
033                for(Integer i=0; i < ProblemaSudoku.posicionesLibres.size();i++){
034                        map.put(ProblemaSudoku.posicionesLibres.get(i), cuadro.getValoresOcupadosEnPos(i).size());
035                }
036                return map;
037        }
038        
039        @Override
040        public Tipo getTipo() {
041                return Tipo.Todas;
042        }
043}
