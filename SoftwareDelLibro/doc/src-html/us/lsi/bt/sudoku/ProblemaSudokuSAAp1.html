001package us.lsi.bt.sudoku;
002
003import java.util.List;
004import java.util.stream.IntStream;
005
006import com.google.common.collect.Lists;
007
008import us.lsi.algoritmos.ProblemaSA;
009import us.lsi.stream.Stream2;
010import us.lsi.common.ParInteger;
011
012public class ProblemaSudokuSAAp1 extends ProblemaSudoku implements ProblemaSA<EstadoSudokuSAAp1, CuadroSudoku, ParInteger> {
013
014        
015        
016        public ProblemaSudokuSAAp1(String nf) {
017                super(nf);
018        }
019
020        @Override
021        public EstadoSudokuSAAp1 getEstadoInicial() {
022                CuadroSudoku cuadro = CuadroSudoku.create();
023                
024                List<List<Integer>> valoresLibresEnX = Lists.newArrayList();
025
026                for (int x = 0; x < ProblemaSudoku.numeroDeFilas; x++) {
027                        List<Integer> ls = Stream2.create(
028                                        IntStream.rangeClosed(1, ProblemaSudoku.numeroDeFilas)
029                                                        .boxed()).toList();
030                        ls.removeAll(cuadro.getValoresOcupadosEnX().get(x));
031                        valoresLibresEnX.add(ls);
032                }
033                
034                for(int x=0; x < ProblemaSudoku.numeroDeFilas; x++){
035                        int a = 0;
036                        for (Integer p: ProblemaSudoku.posicionesLibresEnX.get(x)) {
037                                        cuadro.setInfo(p, valoresLibresEnX.get(x).get(a));
038                                        a++;
039                                }
040                        }
041                return EstadoSudokuSAAp1.create(cuadro);
042        }
043
044        
045        
046}
