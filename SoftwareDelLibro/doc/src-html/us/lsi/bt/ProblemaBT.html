005/**
006 * <p>Interface que debe implementar un problema para ser resuelto por la técnica de Backtracking </p>
007 * 
008 * <p>La documentación puede encontarse en el: <a href="../../../document/Tema15.pdf" target="_blank">Tema15</a></p>
009 * 
010 * @author Miguel Toro
011 *
012 * @param <S> Tipo de la Solución
013 * @param <A> Tipo de la Alternativa
014 */
015public interface ProblemaBT<S, A> {
016        
017        public enum Tipo{Min,Max,Todas}
018        
019        
020        /**
021         * @return El tipo del problema: minimiza, maximiza o busca un valor que involucra a todas las alternativas
022         *
023         * 
024         **/
025        Tipo getTipo(); 
026        /**
027         * @return El estado inicial
028         */
029        EstadoBT<S,A> getEstadoInicial();       
030}
