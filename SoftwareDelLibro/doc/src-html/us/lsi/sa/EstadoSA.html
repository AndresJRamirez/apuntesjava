001package us.lsi.sa;
002
003/**
004 * 
005 * <p>La documentación puede encontarse en el: <a href="../../../document/Tema16.pdf" target="_blank">Tema16</a></p>
006 * 
007 * @author Miguel Toro
008 *
009 * @param <E> Tipo del Estado 
010 * @param <S> Tipo de la solución del problema
011 * @param <A> Tipo de las alternativas
012 */
013public interface EstadoSA<E extends EstadoSA<E,S,A>,S,A> {      
014        /**
015         * @param a - Alternativa elegida
016         * @return Siguiente estado si se escoge la alternativa a
017         */
018        E next(A a);    
019        /**
020         * @return La alternativa escogida en el estado actual. 
021         */
022        A getAlternativa();
023        /**
024         * @return Solución del problema calculada a partir del estado actual
025         */
026        S getSolucion();
027        /**
028         * La condición de parada establece cuando se parará el algortimo. Esta condición se puede establecer a partir
029         * de la propiedades fijadas para el algortimo: número de soluciones encontradas, número de iteraciones, 
030         * valor de la función objetivo, etc.
031         * 
032         * @return Si se cumple la condición del parada del algortimo.
033         */
034        boolean condicionDeParada();
035        /**
036         * @return Valor de la propiedad a minimizar
037         */
038        double getObjetivo(); 
039        /**
040         * @return Una copia del estado 
041         */
042        E copia();
043        /**
044         * @param s En un valor del tipo S
045         * @return Si s es una solcuión del problema
046         */
047        boolean esSolucion(S s);
048}
