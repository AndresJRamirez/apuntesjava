001package us.lsi.sa.reinas;
002
003import java.util.List;
004
005import com.google.common.collect.Lists;
006
007import us.lsi.bt.reinas.Reina;
008import us.lsi.bt.reinas.TableroDeReinas;
009import us.lsi.common.ParInteger;
010import us.lsi.sa.ProblemaSA;
011
012
013
014public class ProblemaReinasSA implements ProblemaSA<EstadoReinasSA,List<Reina>,ParInteger> {
015        
016        public static int numeroDeReinas = 8;
017        
018        @Override
019        public EstadoReinasSA getEstadoInicial() {
020                List<Integer> filasOcupadas = Lists.newArrayList();
021                for (int i = 0; i < numeroDeReinas; i++) {
022                        filasOcupadas.add(i);
023                }
024                TableroDeReinas t = TableroDeReinas.create(filasOcupadas);
025                return EstadoReinasSA.create(t);
026        }
027        
028        public static ProblemaReinasSA create() {
029                return new ProblemaReinasSA();
030        }
031
032        
033
034        
035}
