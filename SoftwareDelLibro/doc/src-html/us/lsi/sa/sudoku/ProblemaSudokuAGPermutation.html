001package us.lsi.sa.sudoku;
002
003import java.util.List;
004import java.util.stream.Collectors;
005import java.util.stream.IntStream;
006
007import com.google.common.collect.HashMultiset;
008import com.google.common.collect.Multiset;
009
010import us.lsi.ag.ProblemaAGIndex;
011import us.lsi.ag.agchromosomes.IndexChromosome;
012import us.lsi.common.Lists2;
013
014
015public class ProblemaSudokuAGPermutation implements ProblemaAGIndex<CuadroSudoku> {     
016        
017
018        @Override
019        public Integer getObjectsNumber() {
020                return CuadroSudoku.numPosicionesLibres;
021        }
022
023        @Override
024        public Double fitnessFunction(IndexChromosome cr) {
025                List<Integer> d = cr.decode();
026                CuadroSudoku c = CuadroSudoku.create(d);
027                Double r = 1.*c.getObjetivoMax();
028                return r;
029        }
030
031        @Override
032        public CuadroSudoku getSolucion(IndexChromosome cr) {
033                List<Integer> d = cr.decode();
034                CuadroSudoku c = CuadroSudoku.create(d);
035                return c;
036        }
037        
038        @Override
039        public List<Integer> getNormalSequence() {
040                 Multiset<Integer> ms = HashMultiset.create();
041                 ms.addAll(CuadroSudoku.casillasOcupadas.values());
042                 List<Integer> r = IntStream.rangeClosed(1,CuadroSudoku.numeroDeFilas)
043                                .boxed()
044                                .map(x->Lists2.nCopias(CuadroSudoku.numeroDeFilas-ms.count(x),x).stream())
045                                .flatMap(x->x)
046                                .collect(Collectors.toList());
047                 return r;
048        }
049
050}
