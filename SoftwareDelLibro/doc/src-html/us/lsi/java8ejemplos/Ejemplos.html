034        public static <T> List<T> ejemploA(Stream<T> st){
035                return st.collect(Collectors.<T>toList());
036        }

038        public static List<Double> ejemploB(List<Punto2D> ls){
039                return ls.stream().map(x->x.getX()).collect(Collectors.toList());
040        }
041        
042        public static boolean ejemploC(List<Integer> ls){
043                return ls.stream().allMatch(x->x%2==1);
044        }
045        
046        public static boolean ejemploD(List<Integer> ls){
047                return ls.stream().anyMatch(x->x%2==1 && Math2.esPrimo(x));
048        }
049        
050        public static Double ejemploE(List<Double> ls){
051                return ls.stream().mapToDouble(x->x).sum();
052        }
053        
054        public static Integer ejemploF(List<Integer> ls){
055                return ls.stream().mapToInt(x->x*x).sum();
056        }
057        
058        public static Double ejemploG(List<Double> ls, Double umbral){
059                return ls.stream().filter(x-> umbral > x).findFirst().get();
060        }
061        
062        public static List<Punto2D> ejemploH(List<Punto2D> ls){
063                return ls.stream().map(x->Punto2D.create(-x.getX(),x.getY())).collect(Collectors.toList());
064        }

066        public static Punto2D ejemploI(List<Punto2D> ls){
067                return ls.stream().min(Comparator.comparing(Punto2D::getX)).get();
068        }
069        
070        public static Long ejemploJ(List<Punto2D> ls){
071                return ls.stream().filter(x->x.getX()>=0. && x.getY()>=0.).count();
072                
073        }

075        public static List<Punto2D> ejemploK(Punto2D[] ls){
076                return Arrays.stream(ls).collect(Collectors.toList());
077        }

079        public static void ejemploL(String fileOut, Integer limit){
080                Stream2.iterate(1, x->Math2.siguientePrimo(x)).whileIncluded(x-> x<=limit).mapToStr(x->x.toString()).toFile(fileOut);
081        }
082        
083        public static Stream<Integer> ejemploM(Stream<List<Integer>> st){
084                return st.flatMap(x->x.stream());
085        }
086        
087        public static void ejemploN(String fileOut, Integer limit){
088                Stream2.iterate(1, x->Math2.siguientePrimo(x)).whileIncluded(x-> x<=limit).map(x->x*x).mapToStr(x->x.toString()).toFile(fileOut);
089        }
090        
091        public static IntStream ejemploO(String fileIn){
092                return Stream2.fromFile(fileIn).mapToInt(x-> new Integer(x));
093        }
094        
095        public static Stream<Punto2D> ejemploP(Integer limit){
096                return Stream2.iterate(1L, x->Math2.siguientePrimo(x)).whileIncluded(x-> x<=limit).<Punto2D>map(x->Punto2D.create((double)x, (double)x));
097        }
098        
099        public static void ejemploQ(String fileIn, String fileOut, Fecha c1, Fecha c2){
100                Preconditions.checkArgument(c2.compareTo(c1) > 0);
101                Stream2.fromFile(fileIn).map(x->x.split(","))
102                                .<Fecha>map(x-> Fecha.create(new Integer(x[0]),new Integer(x[1]),new Integer(x[2])))
103                                .filter(x->c1.compareTo(x)< 0 && c2.compareTo(x)>0)
104                                .sorted()
105                                .mapToStr(x->x.toString())
106                                .toFile(fileOut);
107        }
108        
109        public static IntStream ejemploR(String fileIn){
110                return Stream2.fromFile(fileIn).map(x->x.split(",")).flatMap(x->Arrays.stream(x)).mapToInt(x-> new Integer(x));
111        }

113        public enum Cuadrante{PRIMER_CUADRANTE, SEGUNDO_CUADRANTE, TERCER_CUADRANTE, CUARTO_CUADRANTE};
114        
115        public static Cuadrante getCuadrante(Punto2D p){
116                return null;
117        }
118        
119        public static Multimap<Cuadrante,Punto2D> ejemploS1(Stream<Punto2D> st){
120                return st.map(x->Entry.<Cuadrante,Punto2D>create(getCuadrante(x),x)).collect(Collectors2.toListMultimap());
121        }
122        
123        public static Map<Cuadrante,List<Punto2D>> ejemploT1(Stream<Punto2D> st){
124                return st.collect(Collectors.groupingBy(Ejemplos::getCuadrante));
125        }
126        
127        public static Map<Cuadrante,Double> ejemploU(Stream<Punto2D> st){
128                return st.collect(Collectors.groupingBy(Ejemplos::getCuadrante, 
129                                                        Collectors.<Punto2D,Double>reducing(0.,x->x.getX(),(x,y)->x+y)));
130        }
131                
132        public static Multiset<Cuadrante> ejemploV(Stream<Punto2D> st){
133                return st.map(Ejemplos::getCuadrante).collect(Collectors2.toMultiset());        
134        }
135        
136        public static Map<Cuadrante,Long> ejemploW(Stream<Punto2D> st){
137                return st.collect(Collectors.groupingBy(Ejemplos::getCuadrante, Collectors.counting()));
138        }

141        public static Long ejemploX(String s){
142                return s.chars().filter(x->Character.isLowerCase(x)).count();
143        }

145        public static String ejemploY(List<String> ls){
146                return ls.stream().max(Comparator.comparing(Ejemplos::ejemploX)).get();
147        }

149        public static IntSummaryStatistics ejemploZ(List<Integer> ls){
150                return ls.stream().collect(Collectors.summarizingInt(x->x));
