001package us.lsi.java8ejemplos;
002
003import java.math.BigInteger;
004import java.util.*;
005import java.util.function.*;
006import java.util.stream.*;
007
008import us.lsi.math.MutableBigInteger;
009import us.lsi.stream.*;
010
011
012
013/**
014 *
015 * @author Miguel Toro
016 */
017public class Java8Ejemplos {
018
019        private static final Consumer<Integer> imprimeEntero = x -> {
020                System.out.println(x);
021        };
022
023        public static void ejemplo1() {
024
025                Double r = DoubleStream.iterate(7.0, x -> x * x).filter(x -> x > 3000)
026                                .findFirst().getAsDouble();
027
028                System.out.println("r= " + r);
029
030        }
031
032        public static void ejemplo2() {
033
034                Stream<Integer> s2 = IntStream.range(23, 550).boxed();
035
036            s2.forEach(x -> {
037                        System.out.println(x);
038                });
039
040        }
041
042        public static void ejemplo3() {
043
044                Stream2.create(IntStream.range(50, 60).boxed())
045                                .concat(Stream2.create(IntStream.range(2, 20).boxed()))
046                                .forEach(imprimeEntero);
047        }
048
049        public static void ejemplo4() {
050
051                IntStream.range(0, 5).boxed().forEach(imprimeEntero);
052                System.out.println("========");
053
054                IntStream.range(10, 15).boxed().forEach(imprimeEntero);
055
056                System.out.println("========");
057
058                Stream2.create(IntStream.range(0, 5).boxed())
059                                .<Integer, Boolean, Integer> joint(
060                                                Stream2.create(IntStream.range(10, 15).boxed()),
061                                                x -> x % 2 == 0, x -> x % 3 == 0, (x, y) -> x + y)
062                                .forEach(imprimeEntero);
063
064        }
065
066        public static void ejemplo5() {
067
068                final MutableBigInteger num = new MutableBigInteger(BigInteger.ONE);
069
070                final StringJoiner sj = new StringJoiner("\n");
071
072                System.out.println("\t" + "\t" + "Valores\n" + sj.toString());
073                System.out.println(num.toString());
074        }
075
076}
