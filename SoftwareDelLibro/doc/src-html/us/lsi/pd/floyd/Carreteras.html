001package us.lsi.pd.floyd;
002
003import org.jgrapht.graph.DefaultWeightedEdge;
004
005import us.lsi.graphs.StringEdgeFactory;
006import us.lsi.pd.floyd.Ciudades.Ciudad;
007
008public class Carreteras implements StringEdgeFactory<Ciudad, Carreteras.Carretera> {
009
010        /**
011         * 
012         */
013        
014        public Carreteras() {
015                super();
016                // TODO Auto-generated constructor stub
017        }
018
019        private static Carreteras r = new Carreteras();
020        
021        @Override
022        public Carretera createEdge(Ciudad c1, Ciudad c2) {
023                // TODO Auto-generated method stub
024                return new Carretera();
025        }
026
027        @Override
028        public Carretera createEdge(Ciudad v1, Ciudad v2, String[] formato) {
029                // TODO Auto-generated method stub
030                return new Carretera(formato);
031        }
032
033        public static StringEdgeFactory<Ciudad, Carretera> getFactory(){
034                return r;
035        }
036        
037        public static class Carretera extends DefaultWeightedEdge {
038
039                /**
040                 * 
041                 */
042                private static final long serialVersionUID = 1L;
043                /**
044                 * 
045                 */
046                private static int ultimo = 0;
047                private String nombre;
048                private int id;
049                
050                public Carretera(String[] nombre) {
051                        super();
052                        this.nombre = nombre[0];
053                        this.id = ultimo;
054                        ultimo++;
055                }
056
057                public Carretera() {
058                        super();
059                        this.nombre = "";
060                        this.id = ultimo;
061                        ultimo++;
062                }
063                
064                @Override
065                public String toString() {
066                        return this.nombre+","+getWeight();
067                }
068
069                @Override
070                public int hashCode() {
071                        final int prime = 31;
072                        int result = 1;
073                        result = prime * result + id;
074                        result = prime * result
075                                        + ((nombre == null) ? 0 : nombre.hashCode());
076                        return result;
077                }
078
079                @Override
080                public boolean equals(Object obj) {
081                        if (this == obj)
082                                return true;
083                        if (obj == null)
084                                return false;
085                        if (getClass() != obj.getClass())
086                                return false;
087                        Carretera other = (Carretera) obj;
088                        if (id != other.id)
089                                return false;
090                        if (nombre == null) {
091                                if (other.nombre != null)
092                                        return false;
093                        } else if (!nombre.equals(other.nombre))
094                                return false;
095                        return true;
096                }       
097        }
098}
