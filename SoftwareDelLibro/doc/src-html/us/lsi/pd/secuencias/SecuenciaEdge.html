001package us.lsi.pd.secuencias;
002
003import us.lsi.graphs.SimpleEdge;
004import us.lsi.pd.secuencias.Secuencia.Accion;
005
006public class SecuenciaEdge extends SimpleEdge<Secuencia> {
007
008        /**
009         * 
010         */
011        private static final long serialVersionUID = 1L;
012
013        public static SecuenciaEdge create(Secuencia c1, Secuencia c2, Accion a) {
014                return new SecuenciaEdge(c1, c2, a);
015        }
016
017        private Accion a;
018        private Double valor;
019        
020        private SecuenciaEdge(Secuencia c1, Secuencia c2, Accion a) {
021                super(c1, c2);
022                this.a = a;
023                switch(a){
024                case Cambia: valor = 1.; break;
025                case Elimina: valor = 1.; break;
026                case Añade: valor = 1.; break;
027                case Avanza: valor = 0.; break;
028                case Nada: valor = 0.;
029                }
030        }
031
032        public Accion getAccion() {
033                return a;
034        }
035
036        public Double getValor() {
037                return valor;
038        }
039
040        @Override
041        public int hashCode() {
042                final int prime = 31;
043                int result = super.hashCode();
044                result = prime * result + ((a == null) ? 0 : a.hashCode());
045                result = prime * result + ((valor == null) ? 0 : valor.hashCode());
046                return result;
047        }
048
049        @Override
050        public boolean equals(Object obj) {
051                if (this == obj)
052                        return true;
053                if (!super.equals(obj))
054                        return false;
055                if (!(obj instanceof SecuenciaEdge))
056                        return false;
057                SecuenciaEdge other = (SecuenciaEdge) obj;
058                if (a != other.a)
059                        return false;
060                if (valor == null) {
061                        if (other.valor != null)
062                                return false;
063                } else if (!valor.equals(other.valor))
064                        return false;
065                return true;
066        }
067
068        @Override
069        public String toString() {
070                return "SecuenciaEdge [a=" + a + ", valor=" + valor + ", getSource()="
071                                + getSource() + ", getTarget()=" + getTarget() + "]";
072        }       
073
074}
