001package us.lsi.pd.secuencias;
002
003import java.util.Arrays;
004import java.util.Set;
005import java.util.stream.Collectors;
006
007import us.lsi.graphs.VirtualVertex;
008
009
010public class Secuencia implements VirtualVertex<Secuencia, SecuenciaEdge> {
011
012        public static Secuencia create(String cadena, int index) {
013                return new Secuencia(cadena, index);
014        }
015
016        public static Secuencia create(String origen) {
017                return new Secuencia(origen, 0);
018        }
019        
020        private String cadena;
021        private Integer tamCadena;
022        private int index;
023        public static String transformada;
024        public static Integer tamTransformada;
025        
026        
027        private Secuencia(String cadena, int index) {
028                super();
029                this.cadena = cadena;
030                this.index = index;
031                this.tamCadena = this.cadena.length();
032        }
033
034        public static enum Accion{Añade, Cambia, Elimina, Avanza, Nada};
035        
036        
037        public Integer getTamCadena() {
038                return tamCadena;
039        }
040
041        public int getIndex() {
042                return index;
043        }
044
045        @Override
046        public Set<Secuencia> getNeighborListOf() {
047                return this.edgesOf().stream().map(x->x.getTarget()).collect(Collectors.toSet());
048        }
049
050        @Override
051        public Set<SecuenciaEdge> edgesOf() {
052                return Arrays.stream(Secuencia.Accion.values())
053                                .filter(x->testAccion(x))
054                                .map(x->SecuenciaEdge.create(this,getNeighbor(x), x))
055                                .collect(Collectors.toSet());
056        }
057
058        @Override
059        public boolean isNeighbor(Secuencia e) {
060                return this.getNeighborListOf().contains(e);
061        }
062
063        public Secuencia getNeighbor(Accion a){
064                StringBuilder s = new StringBuilder(cadena);
065                int index = this.index;
066                switch(a){
067                case Cambia: s.setCharAt(index, transformada.charAt(index));  index++; break;
068                case Elimina: s = s.deleteCharAt(index); break;
069                case Añade: s = s.append(transformada.charAt(index));  index++ ; break;
070                case Avanza: index++; break;
071                case Nada:  break;
072                }
073                String r = s.toString();
074                return Secuencia.create(r,index);
075        }
076        
077
078        private boolean testAccion(Accion a){
079                boolean r = true;
080                switch(a){
081                case Cambia: r = index < tamCadena &&  index < tamTransformada? true : false ; break;
082                case Elimina: r = index < tamCadena? true : false ; break;
083                case Añade: r = index == tamCadena &&  index < tamTransformada? true : false ; break;
084                case Avanza:  r = index < tamCadena &&  index < tamTransformada && cadena.charAt(index) == transformada.charAt(index) ? true : false ; break;
085                case Nada:  r = false; break;
086                }
087                return r;
088        }
089
090        @Override
091        public int hashCode() {
092                final int prime = 31;
093                int result = 1;
094                result = prime * result + ((cadena == null) ? 0 : cadena.hashCode());
095                result = prime * result + index;
096                return result;
097        }
098
099        @Override
100        public boolean equals(Object obj) {
101                if (this == obj)
102                        return true;
103                if (obj == null)
104                        return false;
105                if (!(obj instanceof Secuencia))
106                        return false;
107                Secuencia other = (Secuencia) obj;
108                if (cadena == null) {
109                        if (other.cadena != null)
110                                return false;
111                } else if (!cadena.equals(other.cadena))
112                        return false;
113                if (index != other.index)
114                        return false;
115                return true;
116        }
117
118        @Override
119        public String toString() {
120                return "Secuencia [cadena=" + cadena + ", index=" + index + "]";
121        }
122        
123        
124}
