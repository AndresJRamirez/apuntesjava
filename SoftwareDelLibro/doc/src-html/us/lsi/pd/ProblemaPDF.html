013 */
014public interface ProblemaPDF<S, A, T extends Comparable<? super T>> extends ProblemaPD<S, A, T> {
015        /**
016         * @see AlgoritmoPD.Tipo
017         * @constraint getObjetivoEstimado(a) &gt; = getObjetivo()  si AlgoritmoPD.tipo = Max
018         * @constraint getObjetivoEstimado(a) &lt; = getObjetivo()  si AlgoritmoPD.tipo = Min
019         * @param a - Alternativa elegida
020         * @return Valor estimado para propiedad objetivo del problema inicial 
021         * asumiendo que se ha alcanzado el problema actual y se  va a seguir la alaternativa a
022         */
023        T getObjetivoEstimado(A a);
024        /**
025         * @pre esCasoBase()
026         * @return Valor de propiedad objetivo del problema inicial 
027         */
028        T getObjetivo();
029}
