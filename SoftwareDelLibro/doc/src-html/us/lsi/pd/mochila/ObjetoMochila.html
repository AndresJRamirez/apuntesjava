001package us.lsi.pd.mochila;
002
003/**
004 * <p> Esta clase implementa el tipo ObjetoMochila.</p>
005 * <p> Las propiedades de estos problemas son: </p>
006 * <ul>
007 * <li> Código
008 * <li> Valor
009 * <li> Peso
010 * <li> Número máximo de unidades
011 * </ul> 
012 * 
013 * 
014 * 
015 * @author Miguel Toro
016 *
017 */
018public class ObjetoMochila implements Comparable<ObjetoMochila>{
019        
020        public static ObjetoMochila create(Integer valor,Integer peso, Integer countMax) {
021                return new ObjetoMochila(valor, peso, countMax);
022        }
023
024        /**
025         * @param s Una línea de un fichero de texto
026         * @return Construye un objeto mochila a partir de una línea de un fichero
027         */
028        public static ObjetoMochila create(String s) {
029                return new ObjetoMochila(s);
030        }
031        
032        private static Integer nCodigo = 0;
033        
034        private Integer codigo;
035        private Integer valor;
036        private Integer peso;
037        private Integer numMaxDeUnidades;
038        
039        ObjetoMochila(Integer valor, Integer peso, Integer countMax){
040                this.codigo = nCodigo;
041                nCodigo++;
042                this.valor = valor;
043                this.peso = peso;
044                this.numMaxDeUnidades = countMax;
045        }       
046        ObjetoMochila(String s){                
047                String[] v = s.split("[ ,]");
048                Integer ne = v.length;
049                if(ne != 3) throw new IllegalArgumentException("Formato no adecuado en línea  "+s);     
050                valor = new Integer(v[0]);
051                peso = new Integer(v[1]);
052                numMaxDeUnidades = new Integer(v[2]);
053                this.codigo = nCodigo;
054                nCodigo++;
055        }       
056        public Integer getPeso() {
057                return peso;
058        }
059        public Integer getValor() {
060                return valor;
061        }
062        public Integer getCodigo() {
063                return codigo;
064        }
065        public Integer getNumMaxDeUnidades() {
066                return numMaxDeUnidades;
067        }
068        
069        @Override
070        public int compareTo(ObjetoMochila o) {
071                int r = getRatioValorPeso().compareTo(o.getRatioValorPeso());
072                if(r == 0){
073                        r = getCodigo().compareTo(o.getCodigo());
074                }
075                return r;
076        }       
077        public Double getRatioValorPeso() {
078                return ((double)valor)/peso;
079        }
080        
081        @Override
082        public String toString() {
083                return "<"+codigo+","+valor+","+peso+","+numMaxDeUnidades+">";
084        }
085
086        @Override
087        public int hashCode() {
088                final int prime = 31;
089                int result = 1;
090                result = prime * result + ((codigo == null) ? 0 : codigo.hashCode());
091                return result;
092        }
093
094        @Override
095        public boolean equals(Object obj) {
096                if (this == obj)
097                        return true;
098                if (obj == null)
099                        return false;
100                if (!(obj instanceof ObjetoMochila))
101                        return false;
102                ObjetoMochila other = (ObjetoMochila) obj;
103                if (codigo == null) {
104                        if (other.codigo != null)
105                                return false;
106                } else if (!codigo.equals(other.codigo))
107                        return false;
108                return true;
109        }
110        
111        
112        
113}
