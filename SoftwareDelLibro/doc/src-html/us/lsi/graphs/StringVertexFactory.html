004/**
005 * Tipo que deben los vértices de un grafo que queremos leer de un fichero cuyas líneas
006 * representan una arista. Las líneas deben ser de la forma
007 * 
008 * s1,v1,v2,v3
009 * 
010 * Dónde s1 es un identificador únicos del vértice y v1, ... propiedades de la arista
011 * 
012 * @author Miguel Toro
013 *
014 * @param <V> El tipo de los vértices
015 */
016public interface StringVertexFactory<V> {
017        /**
018         * @param formato Los campos de la línea correspondiente: s1,v1,v2,v3
019         * @return El vértice creado
020         */
021        public V createVertex(String[] formato);
022}
