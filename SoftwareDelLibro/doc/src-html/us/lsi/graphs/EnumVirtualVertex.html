001package us.lsi.graphs;
002
003import java.util.List;
004import java.util.Set;
005import java.util.stream.Collectors;
006
007/**
008 * @author Miguel Toro
009 *
010 * @param <A> Tipo de la acción
011 * @param <V> Tipo del vértice
012 * 
013 * <a> Tipo adecuado para modelar un vértice de un grafo virtual simple cuyas aristas están 
014 * definidas por un conjunto finito de acciones codificadas en un tipo enumerado A </a>
015 */
016public abstract class EnumVirtualVertex<A extends Enum<A>, V extends EnumVirtualVertex<A,V>> 
017                        implements VirtualVertex<V, SimpleEdge<V>> {
018
019        public EnumVirtualVertex() {
020        }
021        
022        /**
023         * @return Si es un valor válido del tipo
024         */
025        public abstract boolean isValid();
026        /**
027         * @param a Una acción
028         * @return El vecino tras tomar esa acción
029         */
030        public abstract V neighbor(A a);
031        /**
032         * @param a Una acción
033         * @return Si la acción es aplicable en este vértice
034         */
035        public abstract boolean isApplicable(A a);
036        /**
037         * <a>Para ser implementado por el subtipo </a>
038         * @param v Vértice opuesto
039         * @return Devuelve una arista del vértice actual al opuesto
040         */
041        protected abstract SimpleEdge<V> createEdgeTo(V v);
042        
043        /**
044         * Para ser implementado por el subtipo
045         * @return Lista de valores del tipo enumerado A
046         */
047        protected abstract List<A> values();
048        private Set<V> neighbors = null;
049        private Set<SimpleEdge<V>> edges = null;
050
051        @Override
052        public Set<V> getNeighborListOf() {
053                if (neighbors == null) {
054                        neighbors = values().stream().filter(x -> this.isApplicable(x))
055                                        .map(x -> this.neighbor(x)).collect(Collectors.toSet());
056                }
057                return neighbors;
058        }
059
060        @Override
061        public Set<SimpleEdge<V>> edgesOf() {
062                if (edges == null) {
063                        edges = values().stream().filter(x -> this.isApplicable(x))
064                                        .map(x -> createEdgeTo(this.neighbor(x)))
065                                        .collect(Collectors.toSet());
066                }
067                return edges;
068        }
069
070        @Override
071        public boolean isNeighbor(V e) {
072                return this.getNeighborListOf().contains(e);
073        }
074
075}
