001package us.lsi.graphs;
002
003
004import java.io.*;
005import java.util.*;
006
007import org.jgrapht.*;
008import org.jgrapht.ext.*;
009
010/**
011 * Clase adecuada para salvar un grafo en un fichero en un formato gv.
012 * 
013 * @author Miguel Toro
014 *
015 */
016public class GraphsFileExporter {       
017        
018        
019        public static <V, E> void saveFile(Graph<V, E> graph, String s, VertexNameProvider<V> vertexIDProvider,
020            VertexNameProvider<V> vertexLabelProvider,
021            EdgeNameProvider<E> edgeLabelProvider,
022            ComponentAttributeProvider<V> vertexAttributeProvider,
023            ComponentAttributeProvider<E> edgeAttributeProvider){
024                File f = new File(s);
025                Writer wr = null;
026                try {
027                        wr = new PrintWriter(f);
028                } catch (FileNotFoundException e) {
029                        throw new IllegalArgumentException("No se ha podido crear el fichero "+s);
030                }
031                DOTExporter<V,E> de = new DOTExporter<V,E>(vertexIDProvider,
032                vertexLabelProvider,
033                edgeLabelProvider,
034                vertexAttributeProvider,
035                edgeAttributeProvider);
036                de.export(wr, graph);
037                try {
038                        wr.close();
039                } catch (IOException e) {
040                        throw new IllegalArgumentException("No se ha podido cerrar el fichero "+s);
041                }
042        }
043        
044        /**
045         * Se construye un fichero con información asociada a vértices y 
046         * aristas dada por sus respectivas toString()
047         * 
048         * @param graph Grafo
049         * @param file Fichero
050         * @param <V> El tipo de los vértices
051         * @param <E> El tipo de las aristas
052         */
053        public static <V, E> void saveFile(Graph<V, E> graph, String file){
054                saveFile(graph, file, new StringNameProvider<V>(),null,null,null,null);
055        }
056        
057        /**
058         * Se construye un fichero con información asociada a vértices y 
059         * aristas dada por sus respectivas toString()
060         * 
061         * Los vértices especiales tienen iconos diferentes, 
062         * igualmente ocurre con las aristas especiales
063         * 
064         * @param graph Grafo 
065         * @param file Fichero
066         * @param specialVertexSet Vértices especiales
067         * @param specialEdgeSet Aristas especiales 
068         * @param <V> El tipo de los vértices
069         * @param <E> El tipo de las aristas
070         */
071        public static <V, E> void saveFile(Graph<V, E> graph, String file, Set<V> specialVertexSet, Set<E> specialEdgeSet){
072                GraphsFileExporter.saveFile(graph, file, new StringNameProvider<V>(),
073                                null,null,
074                                new EtiquetasVerticesEspeciales<V>(specialVertexSet),
075                                new EtiquetasAristasEspeciales<E>(specialEdgeSet));
076        }
077        
078        
079        private static class EtiquetasVerticesEspeciales<V> implements ComponentAttributeProvider<V>{
080                
081                private Set<V> set;
082        
083                public EtiquetasVerticesEspeciales(Set<V> set) {
084                        super();
085                        this.set = set;
086                }
087                
088                @Override
089                public Map<String, String> getComponentAttributes(V v) {
090                        Map<String, String> map = new HashMap<>();
091                        if(set.contains(v)){
092                                map.put("style", "dotted");
093                        }
094                        return map;
095                }
096                
097        }
098        
099        private static class EtiquetasAristasEspeciales<E> implements ComponentAttributeProvider<E>{
100                
101                private Set<E> set;
102        
103                public EtiquetasAristasEspeciales(Set<E> set) {
104                        super();
105                        this.set = set;
106                }
107                
108                @Override
109                public Map<String, String> getComponentAttributes(E e) {
110                        Map<String, String> map = new HashMap<>();
111                        if(set.contains(e)){
112                                map.put("style", "dotted");
113                        }
114                        return map;
115                }
116                
117        }
118        
119        
120        
121}
