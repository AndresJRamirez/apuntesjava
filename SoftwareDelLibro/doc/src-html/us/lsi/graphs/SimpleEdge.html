001package us.lsi.graphs;
002
003import org.jgrapht.EdgeFactory;
004
005/**
006 * Un arista simple entre dos vértices de tipo V
007 * El tipo también implementa la factoría de aristas
008 * 
009 * @author Miguel Toro
010 *
011 * @param <V> El tipo de los vértices de la arista
012 */
013public class SimpleEdge<V> implements EdgeFactory<V, SimpleEdge<V>> {
014
015        private V source;
016        private V target;
017        
018        /**
019         * 
020         * @param <E> El tipo de las aristas
021         * @return La factoría de aristas
022         */
023        public static <E> SimpleEdge<E> getFactoria() {
024                return  new SimpleEdge<E>(null,null);
025        }
026
027        protected SimpleEdge(V c1, V c2) {
028                super();
029                this.source = c1;
030                this.target = c2;
031        }
032
033        /**
034         * @return El vértice origen
035         */
036        public V getSource(){
037                return this.source;
038        }
039        
040        /**
041         * @return El vértice destino
042         */
043        public V getTarget(){
044                return this.target;
045        }
046        
047        @Override
048        public String toString() {
049                return this.source+","+this.target;
050        }
051
052        /* (non-Javadoc)
053         * @see org.jgrapht.EdgeFactory#createEdge(java.lang.Object, java.lang.Object)
054         */
055        @Override
056        public SimpleEdge<V> createEdge(V v1, V v2) {
057                return new SimpleEdge<V>(v1,v2);
058        }
059
060        @Override
061        public int hashCode() {
062                final int prime = 31;
063                int result = 1;
064                result = prime * result + ((source == null) ? 0 : source.hashCode());
065                result = prime * result + ((target == null) ? 0 : target.hashCode());
066                return result;
067        }
068
069        @Override
070        public boolean equals(Object obj) {
071                if (this == obj)
072                        return true;
073                if (obj == null)
074                        return false;
075                if (!(obj instanceof SimpleEdge))
076                        return false;
077                SimpleEdge<?> other = (SimpleEdge<?>) obj;
078                if (source == null) {
079                        if (other.source != null)
080                                return false;
081                } else if (!source.equals(other.source))
082                        return false;
083                if (target == null) {
084                        if (other.target != null)
085                                return false;
086                } else if (!target.equals(other.target))
087                        return false;
088                return true;
089        }
090
091        
092        
093        
094}
