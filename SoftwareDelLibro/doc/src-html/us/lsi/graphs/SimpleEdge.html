001package us.lsi.graphs;
002
003/**
004 * <p> Un arista simple entre dos vértices de tipo V </p>
005 * 
006 * @author Miguel Toro
007 *
008 * @param <V> El tipo de los vértices de la arista
009 */
010public class SimpleEdge<V> {
011
012        private V source;
013        private V target;
014        private double weight;
015
016                
017        /**
018         * @param v1 Un vértice
019         * @param v2 Un segundo vértice
020         * @param <V> El tipo de los vértices que une la arista
021         * @return Una arista entre ambos vértices
022         */
023        public static <V> SimpleEdge<V> create(V v1, V v2) {
024                return new SimpleEdge<>(v1,v2);
025        }
026        
027        /**
028         * @param v1 Un vértice
029         * @param v2 Un segundo vértice
030         * @param weight El peso de la arista
031         * @param <V> el tipo de los vértices
032         * @return Una arista entre ambos vértices
033         */
034        public static <V> SimpleEdge<V> create(V v1, V v2, double weight) {
035                return new SimpleEdge<V>(v1, v2, weight);
036        }
037
038
039        protected SimpleEdge(V c1, V c2) {
040                super();
041                this.source = c1;
042                this.target = c2;
043                this.weight = 1.;
044        }
045
046        private SimpleEdge(V c1, V c2, double weight) {
047                super();
048                this.source = c1;
049                this.target = c2;
050                this.weight = weight;
051        }
052        
053        
054        /**
055         * @return El vértice origen
056         */
057        public V getSource(){
058                return this.source;
059        }
060        
061        /**
062         * @return El vértice destino
063         */
064        public V getTarget(){
065                return this.target;
066        }
067        
068        /**
069         * @return El peso asociado a la arista
070         */
071        public double getEdgeWeight(){
072                return this.weight;
073        }
074        
075        
076        @Override
077        public String toString() {
078                return "("+this.source+","+this.target+")";
079        }
080
081        @Override
082        public int hashCode() {
083                final int prime = 31;
084                int result = 1;
085                result = prime * result + ((source == null) ? 0 : source.hashCode());
086                result = prime * result + ((target == null) ? 0 : target.hashCode());
087                return result;
088        }
089
090        @Override
091        public boolean equals(Object obj) {
092                if (this == obj)
093                        return true;
094                if (obj == null)
095                        return false;
096                if (!(obj instanceof SimpleEdge))
097                        return false;
098                SimpleEdge<?> other = (SimpleEdge<?>) obj;
099                if (source == null) {
100                        if (other.source != null)
101                                return false;
102                } else if (!source.equals(other.source))
103                        return false;
104                if (target == null) {
105                        if (other.target != null)
106                                return false;
107                } else if (!target.equals(other.target))
108                        return false;
109                return true;
110        }       
111        
112}
