001package us.lsi.graphs;
002
003import java.util.Set;
004import java.util.function.Predicate;
005
006import org.jgrapht.Graph;
007
008public interface AStarGraph<V, E> extends Graph<V,E> {
009        /**
010         * @param vertex es el vértice actual
011         * @return El peso de vertex
012         */
013        double getVertexWeight(V vertex);
014        /**
015         * @param vertex El vértice actual
016         * @param edgeIn Una arista entrante o incidente en el vértice actual
017         * @param edgeOut Una arista saliente o incidente en el vértice actual
018         * @return el peso asociado al vértice suponiendo las dos aristas dadas dónde la primera puede ser null
019         */
020        double getVertexWeight(V vertex, E edgeIn, E edgeOut);
021        
022        /**
023         * @param actual El vértice actual
024         * @param endVertex El vértice destino
025         * @param goal El predicado que debe cumplir el vértice destino
026         * @param goalSet Un conjunto de vértices destino
027         * @return Una cota inferior del peso del camino desde el vértice actual al destino, 
028         * o desde el vértice actual al vértice que cumple el predicado
029         * o de la suma de los pesos de los caminos mínimos del vértice actual a los objetivos
030         */
031        double getWeightToEnd(V actual, V endVertex, Predicate<V> goal, Set<V> goalSet);
032}
