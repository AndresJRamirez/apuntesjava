001package us.lsi.graphs;
002
003import java.util.List;
004import java.util.Map;
005
006import org.jgrapht.Graph;
007
008import com.google.common.base.Preconditions;
009import com.google.common.collect.Lists;
010import com.google.common.collect.Maps;
011
012public class GraphView<V, E> {
013
014        public static <V, E> GraphView<V, E> create(Graph<V, E> grafo) {
015                return new GraphView<V, E>(grafo);
016        }
017
018        private List<V> vertices;
019        private Graph<V,E> grafo;
020        private Integer numVertices;
021        private Map<V,Integer> map;
022        
023        private GraphView(Graph<V,E> grafo) {
024                this.grafo = grafo;
025                this.vertices = Lists.newArrayList(grafo.vertexSet());
026                this.map = Maps.newHashMap();
027                Integer n = 0;
028                for(V v:vertices){
029                        map.put(v,n);
030                        n++;
031                }
032                this.numVertices = this.vertices.size();
033        }
034
035        public List<V> getVertices() {
036                return vertices;
037        }
038
039        public Graph<V, E> getGrafo() {
040                return grafo;
041        }
042
043        public Integer getNumVertices() {
044                return numVertices;
045        }
046
047        public V getVertice(int i) {
048                return vertices.get(i);
049        }
050        
051        public Integer getIndex(V v) {
052                return map.get(v);
053        }
054        
055        public Double getWeight(int i, int j) {
056                Preconditions.checkArgument(this.isEdge(i,j));
057                V v1 = getVertice(i);
058                V v2 = getVertice(j);
059                E cr = grafo.getEdge(v1,v2);
060                return grafo.getEdgeWeight(cr);
061        }
062        
063        public boolean isEdge(int i, int j) {
064                V v1 = getVertice(i);
065                V v2 = getVertice(j);
066                E e = grafo.getEdge(v1,v2);
067                return e==null?false:true;
068        }
069        
070        public E getEdge(int i, int j) {
071                Preconditions.checkArgument(this.isEdge(i,j));
072                V v1 = getVertice(i);
073                V v2 = getVertice(j);
074                E e = grafo.getEdge(v1,v2);
075                return e;
076        }
077}
