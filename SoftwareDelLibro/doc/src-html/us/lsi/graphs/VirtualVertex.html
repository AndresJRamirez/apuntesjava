003import java.util.Set;
004
005/**
006 * <a> El tipo representa un vértice de un grafo virtual no dirigido </a>
007 * 
008 * @author Miguel Toro
009 *
010 * @param <V> Tipo de los vértices
011 * @param <E> Tipo de las aristas
012 */
013public interface VirtualVertex<V extends VirtualVertex<V,E>, E extends SimpleEdge<V>> { 
014        /**
015         * @return Si es un valor válido del tipo
016         */
017        boolean isValid();
018        /**
019         * @return Conjunto de los vértices vecinos
020         */
021        Set<V> getNeighborListOf();
022        /**
023         * @return Conjunto de las aristas hacia los vértices vecinos
024         */
025        Set<E> edgesOf(); 
026        
027        /**
028         * @param v Vértice que se pregunta si es vecino
029         * @return Si el vértice es vecino
030         */
031        default boolean isNeighbor(V v) {
032                return getNeighborListOf().contains(v);
033        }
034

036//   * @return La arista hacia el vértice v
037//       */
038//      E getEdge(V v);

040}
