001package us.lsi.graphs;
002
003import org.jgrapht.EdgeFactory;
004
005/**
006 * Tipo que deben implementar la aristas de un grafo que queremos leer de un fichero cuyas líneas
007 * representa una arista. Las líneas deben ser de la forma
008 * 
009 * s1,s2,v1,v2,v3
010 * 
011 * Dónde s1 y s2 son idenficadores únicos de vértices y v1, ... propiedades de la arista
012 * 
013 * @author Miguel Toro
014 *
015 * @param <V> Tipo de los vértices
016 */
017public interface StringEdgeFactory<V, E> extends EdgeFactory<V, E> {
018        /**
019         * @param v1 Vértice origen
020         * @param v2 Vértice destino
021         * @param formato Propiedades v1,v2, ...
022         * @return La arista creada
023         */
024        public E createEdge(V v1, V v2, String[] formato);
025}
