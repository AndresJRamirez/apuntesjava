001package us.lsi.stream;
002
003import java.util.*;
004import java.util.stream.Collectors;
005
006
007
008
009/**
010 * Es un tipo que asocia claves de tipo K1, K2 con valores de tipo R. Una versión compacta de un Map &lt;K1, Map &lt; K2,R&gt; &gt;.
011 * 
012 * @author Miguel Toro
013 *
014 * @param <K1> Tipo de la primera clave 
015 * @param <K2> Tipo de la segunda clave 
016 * @param <R> Tipo de los valores
017 */
018public class Group2<K1, K2, R> extends HashMap<K1, Map<K2, R>> implements Map<K1, Map<K2, R>> {
019
020        /**
021         * 
022         */
023        private static final long serialVersionUID = 1L;
024
025        public static <K1, K2, R> Group2<K1, K2, R> create(
026                        Map<? extends K1, ? extends Map<K2, R>> m) {
027                return new Group2<K1, K2, R>(m);
028        }
029
030        public String toString(int n){
031                return this
032                                .keySet()
033                                .stream()
034                                .<String>map(x->Grouping.nIndex(n)+x.toString()+"\n"+this.getGroup(x).toString(n+1))
035                                .collect(Collectors.joining("\n"));
036        }
037        
038        public String toString(){
039                return toString(0);
040        }
041
042        public Group1<K2, R> getGroup(K1 x){
043                return Group1.create(this.get(x));
044        }
045
046        protected Group2(Map<? extends K1, ? extends Map<K2, R>> m) {
047                super(m);
048        }
049        
050}
