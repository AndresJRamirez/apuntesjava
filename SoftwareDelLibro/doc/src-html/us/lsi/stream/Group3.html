001package us.lsi.stream;
002
003import java.util.*;
004import java.util.stream.Collectors;
005
006/**
007 * Es un tipo que asocia claves de tipo K1, K2 con valores de tipo R. Una versión compacta de un Map &lt;K1, Map &lt; K2, Map &lt; K3, &gt; &gt; &gt;.
008 * 
009 * @author Miguel Toro
010 *
011 * @param <K1> Tipo de la primera clave 
012 * @param <K2> Tipo de la segunda clave 
013 * @param <K3> Tipo de la tercera clave 
014 * @param <R> Tipo de los valores
015 */
016
017public class Group3<K1, K2, K3, R> extends HashMap<K1, Map<K2, Map<K3, R>>>
018        implements Map<K1, Map<K2, Map<K3, R>>> {
019
020        
021        private static final long serialVersionUID = 1L;
022        
023        public static <K1, K2, K3, R> Group3<K1, K2, K3, R> create(
024                        Map<? extends K1, ? extends Map<K2, Map<K3, R>>> m) {
025                return new Group3<K1, K2, K3, R>(m);
026        }
027
028        public String toString(int n) {
029                return this
030                                .keySet()
031                                .stream()
032                                .<String> map(
033                                                x -> Grouping.nIndex(n) + x.toString() + "\n"
034                                                                + this.getGroup(x).toString(n+1))
035                                .collect(Collectors.joining("\n"));
036        }
037        
038        public String toString(){
039                return toString(0);
040        }
041
042        
043        public Group2<K2, K3, R> getGroup(K1 x) {
044                return Group2.create(this.get(x));
045        }
046
047        protected Group3(Map<? extends K1, ? extends Map<K2, Map<K3, R>>> m) {
048                super(m);
049        }
050}
