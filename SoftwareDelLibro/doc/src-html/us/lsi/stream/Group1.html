001package us.lsi.stream;
002
003import java.util.*;
004import java.util.stream.Collectors;
005
006/**
007 * Es un tipo que asocia claves de tipo K1 con valores de tipo R. Una versión compacta de un Map&lt;K1,R&gt;.
008 * 
009 * @author Miguel Toro
010 *
011 * @param <K1> Tipo de la clave
012 * @param <R> Tipo de los valores
013 */
014public class Group1<K1, R> extends HashMap<K1, R> implements Map<K1, R> {
015
016        /**
017         * 
018         */
019        private static final long serialVersionUID = 1L;
020
021        public static <K1, R> Group1<K1, R> create(Map<? extends K1, ? extends R> m) {
022                return new Group1<K1, R>(m);
023        }
024
025        public String toString(int n) {
026                return this
027                                .keySet()
028                                .stream()
029                                .<String> map(
030                                                x -> Grouping.nIndex(n) + x.toString() + ",   "
031                                                                + this.get(x))
032                                .collect(Collectors.joining("\n"));
033        }
034
035        public String toString() {
036                return toString(0);
037        }
038
039        public R getGroup(K1 x) {
040                return this.get(x);
041        }
042
043        protected Group1(Map<? extends K1, ? extends R> m) {
044                super(m);
045        }
046        
047}
