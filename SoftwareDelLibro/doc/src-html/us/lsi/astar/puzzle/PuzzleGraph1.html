038                        SimpleEdge<EstadoPuzzle> edgeIn,

042
043        @Override
044        public double getWeightToEnd(EstadoPuzzle startVertex,EstadoPuzzle endVertex, Predicate<EstadoPuzzle> goal,
045                        Set<EstadoPuzzle> goalSet) {
046                if(startVertex==null || endVertex==null)
047                        throw new IllegalArgumentException("El vértice inicial y final no pueden ser null");
