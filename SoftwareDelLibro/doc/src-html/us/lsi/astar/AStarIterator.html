055/**
056 * @author Miguel Toro
057 *
058 * @param <V> Tipo de los vértices
059 * @param <E> Tipo de las aristas
060 */
061public class AStarIterator<V, E> extends CrossComponentIterator<V,E, FibonacciHeapNode<AStarIterator.QueueEntry<V, E>>>
062{
063    //~ Instance fields --------------------------------------------------------
064
065        private AStarGraph<V, E> wGraph;
066        private V endVertex;
067        private Predicate<V> goal;
068        private Set<V> goalSet;
069        
070    /**
071     * Priority queue of fringe vertices.
072     */
073    private FibonacciHeap<QueueEntry<V, E>> heap =
074        new FibonacciHeap<QueueEntry<V, E>>();
075
076    /**
077     * Maximum distance to search.
078     */
079    private double radius = Double.POSITIVE_INFINITY;
080
081    private boolean initialized = false;
082
083    //~ Constructors -----------------------------------------------------------
084
085    /**
086     * Crea un iterador para el grafo
087     *
088     * @param g El grafo
089     */
090    public AStarIterator(AStarGraph<V, E> g)
091    {
092        this(g,null,null,null,null, Double.POSITIVE_INFINITY);
093    }
094
095    /**
096     * Crea un nuevo iterador para el gráfico especificado. Es un iterador que continua por el siguiente vértice 
097     * que cuya función f(v) sea la más pequeña posible. La iteración 
098     * se iniciará en el vértice de inicio especificada y se limitará a la 
099     * componente conexa que incluya ese vértice. 
100     *
101     * @param g Grafo 
102     * @param startVertex Vértice Origen
103     * @param endVertex Vértice destino
104     */
105    public AStarIterator(AStarGraph<V, E> g, V startVertex, V endVertex)
106    {
107        this(g, startVertex, endVertex, null, null, Double.POSITIVE_INFINITY);
108    }

111    /**
112     * @param g Grafo
113     * @param startVertex Vértice origen
114     * @param goal Predicado que debe cumplir el vértice objetivo
115     */
116    public AStarIterator(AStarGraph<V, E> g, V startVertex, Predicate<V> goal)
117    {
118        this(g, startVertex, null,goal, null, Double.POSITIVE_INFINITY);
119    }
120    
121    /**
122     * @param g Grafo 
123     * @param startVertex Vértice origen
124     * @param goalSet Conjunto de vértices objetivo
125     */
126    public AStarIterator(AStarGraph<V, E> g, V startVertex, Set<V> goalSet)
127    {
128        this(g, startVertex, null, null, goalSet, Double.POSITIVE_INFINITY);
129    }
130    
131    
132    AStarIterator(AStarGraph<V, E> g, V startVertex, V endVertex, 
133                Predicate<V> goal, Set<V> goalSet, double radius)
134    {
135        super(g, startVertex);
136        this.radius = radius;
137        checkRadiusTraversal(isCrossComponentTraversal());
138        initialized = true;
139        this.wGraph = (AStarGraph<V, E>) g;
140        this.endVertex = endVertex;
141        this.goal =goal;
142        this.goalSet = goalSet;
143    }
144
145    //~ Methods ----------------------------------------------------------------
146
147   
148    public void setCrossComponentTraversal(boolean crossComponentTraversal)
149    {
150        if (initialized) {
151            checkRadiusTraversal(crossComponentTraversal);
152        }
153        super.setCrossComponentTraversal(crossComponentTraversal);
154    }
155
156    /**
157     * La longitud del camino más corto conocido hasta el vértice proporcionado. 
158     * Si el vértice ha sido ya visitado estonces es verdaderamente la más corta;
159     * pero si en otro caso una cota superior de la misma
160     * 
161     *
162     * @param vertex Vértice para el que se busca el camino hasta el origen
163     *
164     * @return Longitud del camino más corto conocido, o Double.POSITIVE_INFINITY si no 
165     * se ha encontrado ningún camino todavía. 
166     * 
167     */
168    public double getShortestPathLength(V vertex)
169    {
170        FibonacciHeapNode<QueueEntry<V, E>> node = getSeenData(vertex);
171
172        if (node == null) {
173            return Double.POSITIVE_INFINITY;
174        }
175
176        return node.getKey();
177    }
178
179    /**
180     * El árbol de recubrimiento que contiene los caminos desde el origen 
181     * a los vértices alcanzados.
182     *
183     * @param vertex Vértice.
184     *
185     * @return La arista que inicia del camino desde el vértice al origen,
186     * o <code>null</code> si el vértice no ha sido alcanzado 
187     * o es el vértice origen
188     * 
189     */
190    public E getSpanningTreeEdge(V vertex)
191    {
192        FibonacciHeapNode<QueueEntry<V, E>> node = getSeenData(vertex);
193
194        if (node == null) {
195            return null;
196        }
197
198        return node.getData().spanningTreeEdge;
199    }
200
201    /**
202     * @see CrossComponentIterator#isConnectedComponentExhausted()
203     */
204    protected boolean isConnectedComponentExhausted()
205    {
206        if (heap.size() == 0) {
207            return true;
208        } else {
209            if (heap.min().getKey() > radius) {
210                heap.clear();
211
212                return true;
213            } else {
214                return false;
215            }
216        }
217    }
218
219    /**
220     * @see CrossComponentIterator#encounterVertex(Object, Object)
221     */
222    protected void encounterVertex(V vertex, E edge)
223    {
224        double shortestPathLength;
225        double initialWeight;
226        
227        if (edge == null) {
228                initialWeight = calculateInitialPathWeightFromOriginToActual(vertex);
229            shortestPathLength = calculateInitialPathWeight(vertex);
230        } else {
231                initialWeight = calculatePathWeightFromOriginToActual(vertex, edge);
232            shortestPathLength = calculatePathWeight(vertex, edge, initialWeight);
233        }
234        FibonacciHeapNode<QueueEntry<V, E>> node = createSeenData(vertex, edge);
235        node.getData().distance = initialWeight;
236        putSeenData(vertex, node);
237        heap.insert(node, shortestPathLength);
238    }
239
240    /**
241     * Override superclass. When we see a vertex again, we need to see if the
242     * new edge provides a shorter path than the old edge.
243     *
244     * @param vertex the vertex re-encountered
245     * @param edge the edge via which the vertex was re-encountered
246     */
247    protected void encounterVertexAgain(V vertex, E edge)
248    {
249        FibonacciHeapNode<QueueEntry<V, E>> node = getSeenData(vertex);
250
251        if (node.getData().frozen) {
252            // no improvement for this vertex possible
253            return;
254        }
255
256        double distanceFromOriginToActual = calculatePathWeightFromOriginToActual(vertex, edge);
257        double candidatePathLength = calculatePathWeight(vertex, edge, distanceFromOriginToActual);
258
259        if (candidatePathLength < node.getKey()) {
260            node.getData().spanningTreeEdge = edge;
261            node.getData().distance = distanceFromOriginToActual;
262            heap.decreaseKey(node, candidatePathLength);
263        }
264    }
265
266    /**
267     * @see CrossComponentIterator#provideNextVertex()
268     */
269    protected V provideNextVertex()
270    {
271        FibonacciHeapNode<QueueEntry<V, E>> node = heap.removeMin();
272        node.getData().frozen = true;
273
274        return node.getData().vertex;
275    }
276    
277    /**
278     * Determina el peso inicial del camino que sólo contiene el vértice dado
279     *
280     * @param vertex the vertex for which to calculate the path length.
281     *
282     * @return calculated path length.
283     */
284    
285    protected double calculateInitialPathWeightFromOriginToActual(V vertex){
286        return wGraph.getVertexWeight(vertex);
287    }
288    
289    /**
290     *  
291     * Determina el peso total del camino que pasa por el vértice dado
292     *
293     * @param vertex the vertex for which to calculate the path length.
294     *
295     * @return calculated path length.
296     */
297    
298    protected double calculateInitialPathWeight(V vertex){
299        return calculateInitialPathWeightFromOriginToActual(vertex)+
300                        wGraph.getWeightToEnd(vertex,endVertex,goal,goalSet);
301    }
302    
303    /**
304     * Determina la longitud del camino hasta vertex via an edge, using the path length for
305     * the opposite vertex.
306     *
307     * @param vertex the vertex for which to calculate the path length.
308     * @param edge the edge via which the path is being extended.
309     *
310     * @return calculated path length.
311     */
312    
313    protected double calculatePathWeightFromOriginToActual(V vertex, E edge){
314        
315        V otherVertex = Graphs.getOppositeVertex(getGraph(), edge, vertex);
316        FibonacciHeapNode<QueueEntry<V, E>> otherEntry = getSeenData(otherVertex);      
317        
318        E edgeBefore = this.getSpanningTreeEdge(otherVertex);
319        
320        double r = otherEntry.getData().distance 
321                                +getGraph().getEdgeWeight(edge)
322                                        +wGraph.getVertexWeight(vertex)
323                                        +wGraph.getVertexWeight(otherVertex, edgeBefore, edge);
324                return r;
325    }
326    
327    /**
328     * Determine la longitud total del camino a través de vertex via an edge, using the path length for
329     * the opposite vertex.
330     *
331     * @param vertex the vertex for which to calculate the path length.
332     * @param edge the edge via which the path is being extended.
333     * @param distance Distancia máxima
334     *
335     * @return calculated path length.
336     */
337    
338    protected double calculatePathWeight(V vertex, E edge, double distance)  { // cambiado de private a protected               
339        return distance+wGraph.getWeightToEnd(vertex,endVertex,goal,goalSet);
340    }
341
342    private void checkRadiusTraversal(boolean crossComponentTraversal)
343    {
344        if (crossComponentTraversal && (radius != Double.POSITIVE_INFINITY)) {
345            throw new IllegalArgumentException(
346                "radius may not be specified for cross-component traversal");
347        }
348    }
349
350    /**
351     * The first time we see a vertex, make up a new heap node for it.
352     *
353     * @param vertex a vertex which has just been encountered.
354     * @param edge the edge via which the vertex was encountered.
355     *
356     * @return the new heap node.
357     */
358    private FibonacciHeapNode<QueueEntry<V, E>> createSeenData(
359        V vertex,
360        E edge)
361    {
362        QueueEntry<V, E> entry = new QueueEntry<V, E>();
363        entry.vertex = vertex;
364        entry.spanningTreeEdge = edge;
365
366        return new FibonacciHeapNode<QueueEntry<V, E>>(entry);
367    }
368
369    //~ Inner Classes ----------------------------------------------------------
370
371    /**
372     * Private data to associate with each entry in the priority queue.
373     */
374    static class QueueEntry<V, E>
375    {
376        /**
377         * Best spanning tree edge to vertex seen so far.
378         */
379        E spanningTreeEdge;
380
381        /**
382         * The vertex reached.
383         */
384        V vertex;
385
386        /**
387         * True once spanningTreeEdge is guaranteed to be the true minimum.
388         */
389        boolean frozen;
390        
391        /**
392         * Distance to origin 
393         */
394        double distance;
395
396        QueueEntry()
397        {
398        }
399    }
400}
