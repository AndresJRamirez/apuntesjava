001package us.lsi.astar.vuelos;
002
003
004public class Vuelo {
005        
006        private String from;
007        private String to;
008        private double horaDeSalida;
009        private double duracion;
010        
011        public static  Vuelo create(String from, String to, String[] s) {
012                return new Vuelo(from, to, new Double(s[2]),new Double(s[3]));
013        }
014        
015        public static Vuelo create(String c1, String c2) {
016                return create(c1,c2,0.0, 1.0);
017        }
018        
019        
020        Vuelo(String from, String to, double horaDeSalida, double duracion) {
021                super();
022                this.from = from;
023                this.to = to;
024                this.horaDeSalida = horaDeSalida;
025                this.duracion = duracion;
026        }
027
028        
029        public Vuelo() {
030                super();
031                // TODO Auto-generated constructor stub
032        }
033
034
035
036        public String getFrom() {
037                return from;
038        }
039
040
041
042        public String getTo() {
043                return to;
044        }
045
046
047
048        public double getHoraDeSalida() {
049                return horaDeSalida;
050        }
051
052
053
054        public double getDuracion() {
055                return duracion;
056        }
057
058        public double getHoraDeLlegada() {
059                return horaDeSalida+duracion;
060        }
061
062        @Override
063        public int hashCode() {
064                final int prime = 31;
065                int result = 1;
066                result = prime * result + ((from == null) ? 0 : from.hashCode());
067                long temp;
068                temp = Double.doubleToLongBits(horaDeSalida);
069                result = prime * result + (int) (temp ^ (temp >>> 32));
070                result = prime * result + ((to == null) ? 0 : to.hashCode());
071                return result;
072        }
073
074        @Override
075        public boolean equals(Object obj) {
076                if (this == obj)
077                        return true;
078                if (obj == null)
079                        return false;
080                if (!(obj instanceof Vuelo))
081                        return false;
082                Vuelo other = (Vuelo) obj;
083                if (from == null) {
084                        if (other.from != null)
085                                return false;
086                } else if (!from.equals(other.from))
087                        return false;
088                if (Double.doubleToLongBits(horaDeSalida) != Double
089                                .doubleToLongBits(other.horaDeSalida))
090                        return false;
091                if (to == null) {
092                        if (other.to != null)
093                                return false;
094                } else if (!to.equals(other.to))
095                        return false;
096                return true;
097        }
098
099        @Override
100        public String toString() {
101                return "Vuelo [from=" + from + ", to=" + to + ", horaDeSalida="
102                                + horaDeSalida + ", duracion=" + duracion + "]";
103        }
104        
105        
106        public static Vuelo create(String from, String to, double timeSalida,double duracion) {
107                return new Vuelo(from, to, timeSalida, duracion);
108        }
109
110        
111
112        
113}
