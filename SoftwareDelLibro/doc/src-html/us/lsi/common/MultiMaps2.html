001package us.lsi.common;
002
003import com.google.common.collect.ArrayListMultimap;
004import com.google.common.collect.HashMultimap;
005import com.google.common.collect.ListMultimap;
006import com.google.common.collect.Multimap;
007import com.google.common.collect.SetMultimap;
008
009public class MultiMaps2 {
010
011        public static <K,V> ListMultimap<K,V> newListMultimap(){
012                return ArrayListMultimap.<K,V>create();
013        }
014        
015        public static <K,V> ListMultimap<K,V> newListMultimap(Multimap<K,V> m){
016                return ArrayListMultimap.create(m);
017        }
018        
019        public static <K, V> ListMultimap<K, V> newListMultimap(@SuppressWarnings("unchecked") Entry<? extends K, ? extends V>... entries){
020                ListMultimap<K, V> m = ArrayListMultimap.<K,V>create();
021                for(Entry<? extends K, ? extends V> e : entries){
022                        m.put(e.key, e.value);
023                }
024                return m;
025        }
026
027        public static <K,V> SetMultimap<K,V> newSetMultimap(){
028                return HashMultimap.<K,V>create();
029        }
030        
031        public static <K,V> SetMultimap<K,V> newSetMultimap(Multimap<K,V> m){
032                return HashMultimap.create(m);
033        }
034        
035        public static <K, V> SetMultimap<K, V> newSetMultimap(@SuppressWarnings("unchecked") Entry<? extends K, ? extends V>... entries){
036                SetMultimap<K, V> m = HashMultimap.<K,V>create();
037                for(Entry<? extends K, ? extends V> e : entries){
038                        m.put(e.key, e.value);
039                }
040                return m;
041        }
042}
