001package us.lsi.common;
002
003import java.util.List;
004import java.util.function.BinaryOperator;
005import java.util.stream.Collectors;
006
007public class NaryExp<R> extends Exp<R> {
008
009        private List<Exp<R>> ops = null;
010        private BinaryOperator<R> operator;
011        private String symbol = null;   
012
013        NaryExp(List<Exp<R>> ops, BinaryOperator<R> operator, String symbol) {
014                super();
015                this.ops = ops;
016                this.operator = operator;
017                this.symbol = symbol;
018        }
019
020        @Override
021        public Integer getArity() {
022                return ops.size();
023        }
024
025        @Override
026        public R eval() {
027                return ops.stream().map(x->x.eval()).collect(Collectors.reducing(this.operator)).get();
028        }
029
030        @Override
031        public NaryExp<R> clone() {
032                List<Exp<R>> ops = this.ops;
033                if(ops!= null) ops = ops.stream().map(x->x.clone()).collect(Collectors.toList());
034                return Exp.createNary(ops, this.operator, this.symbol);
035        }
036
037
038        public List<Exp<R>> getOps() {
039                return ops;
040        }
041
042        public BinaryOperator<R> getOperator() {
043                return operator;
044        }
045
046        public String getSymbol() {
047                return symbol;
048        }
049
050        @Override
051        public String toString() {
052                return "("+ops.stream().map(x->x.toString()).collect(Collectors.joining(this.symbol))+")";
053        }
054
055        public void setOps(List<Exp<R>> ops) {
056                this.ops = ops;
057        }
058
059        
060}
