001package us.lsi.common;
002
003import java.io.BufferedWriter;
004import java.io.FileWriter;
005import java.io.IOException;
006import java.io.PrintWriter;
007import java.util.Arrays;
008
009public class StringExtensions2 {
010        
011        public static void toFile(String s, String file){
012                try {
013                        final PrintWriter f = new PrintWriter(new BufferedWriter(
014                                        new FileWriter(file)));
015                                f.println(s);
016                        f.close();
017                } catch (IOException e) {
018                        throw new IllegalArgumentException(
019                                        "No se ha podido crear el fichero " + file);
020                }
021        }
022        
023        public static String[] toArray(String s, String delim){
024                return Arrays.<String>stream(s.split(delim))
025                                .<String>map((String x) -> x.trim())
026                                .filter((String x) -> x.length() > 0)
027                                .toArray((int x)-> new String[x]);
028        }
029        
030        public static void toConsole(String s){
031                System.out.println(s);
032        }
033        
034}
