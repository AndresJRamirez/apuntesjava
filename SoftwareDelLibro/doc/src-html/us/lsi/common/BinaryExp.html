001package us.lsi.common;
002
003import java.util.function.BiFunction;
004
005
006/**
007 * @author Miguel Toro
008 * 
009 * <p> Expresión binaria
010 *
011 * @param <R> Tipo del resultado 
012 * @param <T1> Tipo del primer operando
013 * @param <T2> Tipo del segundo operando
014 */
015public class BinaryExp<R,T1,T2> extends Exp<R> {
016
017        protected Exp<T1> op1 = null;
018        protected Exp<T2> op2 = null;
019        protected BiFunction<T1,T2,R> operator;
020        protected String symbol = null;
021        
022        BinaryExp(Exp<T1> left, Exp<T2> right, BiFunction<T1,T2,R> operator, String symbol) {
023                super();
024                this.op1 = left;
025                this.op2 = right;
026                this.operator = operator;
027                this.symbol = symbol;
028        }
029
030        BinaryExp(BiFunction<T1,T2,R> operator, String symbol) {
031                super();
032                this.op1 = null;
033                this.op2 = null;
034                this.operator = operator;
035                this.symbol = symbol;
036        }
037        
038        @Override
039        public Integer getArity() {
040                return 2;
041        }
042
043        public Exp<T1> getOp1() {
044                return op1;
045        }
046
047        public void setOp1(Exp<T1> op1) {
048                this.op1 = op1;
049        }
050
051        public Exp<T2> getOp2() {
052                return op2;
053        }
054
055        public void setOp2(Exp<T2> op2) {
056                this.op2 = op2;
057        }
058
059        public void setOp(Exp<T1> op1, Exp<T2> op2) {
060                this.op1 = op1;
061                this.op2 = op2;
062        }
063        
064        public BiFunction<T1,T2,R> getOperator() {
065                return operator;
066        }
067
068        @Override
069        public R eval() {
070                return operator.apply(op1.eval(), op2.eval());
071        }
072        
073        public String toString() {
074                String r = this.symbol;
075                if (op1!=null) {
076                        r = "(" + op1.toString() + this.symbol + op2.toString() + ")";
077                }
078                return r;
079        }
080
081        @Override
082        public Exp<R> clone() {
083                Exp<T1> e1 = this.op1;
084                Exp<T2> e2 = this.op2;
085                if(e1!= null) e1 = e1.clone();
086                if(e2!= null) e2 = e2.clone();
087                return Exp.createBinary(e1, e2, this.operator, this.symbol);
088        }
089}
