001package us.lsi.common;
002
003public class TernaryExp<R,T1, T2, T3> extends Exp<R> {
004
005        public static interface TriFunction<S1,S2,S3,T> {
006                T apply(S1 op1,S2 op2,S3 op3);
007        }
008        
009        protected Exp<T1> op1 = null;
010        protected Exp<T2> op2 = null;
011        protected Exp<T3> op3 = null;
012        protected TriFunction<T1,T2,T3,R> operator;
013        protected String symbol = null;
014        
015        TernaryExp(Exp<T1> op1, Exp<T2> op2, Exp<T3> op3, TriFunction<T1,T2,T3,R> operator, String symbol) {
016                super();
017                this.op1 = op1;
018                this.op2 = op2;
019                this.op3 = op3;
020                this.operator = operator;
021                this.symbol = symbol;
022        }
023
024        TernaryExp(TriFunction<T1,T2,T3,R> operator, String symbol) {
025                super();
026                this.op1 = null;
027                this.op2 = null;
028                this.op2 = null;
029                this.operator = operator;
030                this.symbol = symbol;
031        }
032        
033        @Override
034        public Integer getArity() {
035                return 3;
036        }
037
038        public Exp<T1> getOp1() {
039                return this.op1;
040        }
041
042        public void setOp1(Exp<T1> op1) {
043                this.op1 = op1;
044        }
045
046        public Exp<T2> getOp2() {
047                return this.op2;
048        }
049
050        public void setOp2(Exp<T2> op2) {
051                this.op2 = op2;
052        }
053
054        public void setOp(Exp<T1> op1, Exp<T2> op2, Exp<T3> op3) {
055                this.op1 = op1;
056                this.op2 = op2;
057                this.op3 = op3;
058        }
059        
060        public TriFunction<T1,T2,T3,R> getOperator() {
061                return operator;
062        }
063
064        @Override
065        public R eval() {
066                return operator.apply(op1.eval(), op2.eval(),op3.eval());
067        }
068        
069        public String toString() {
070                String r = this.symbol;
071                if (this.op1!=null) {
072                        r = this.symbol + "(" + this.op1.toString() + "," + this.op2.toString() + "," + this.op3.toString() + ")";
073                }
074                return r;
075        }
076
077        @Override
078        public Exp<R> clone() {
079                Exp<T1> e1 = this.op1;
080                Exp<T2> e2 = this.op2;
081                Exp<T3> e3 = this.op3;
082                if(e1!= null) e1 = e1.clone();
083                if(e2!= null) e2 = e2.clone();
084                if(e3!= null) e3 = e3.clone();
085                return Exp.createTernary(e1, e2, e3, this.operator,
086                                this.symbol);
087        }
088        
089        @Override
090        public UnaryExpS<R> asUnary() {
091                throw new IllegalStateException("Not a subtype");
092        }
093
094}
