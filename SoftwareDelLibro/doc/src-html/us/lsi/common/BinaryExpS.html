001package us.lsi.common;
002
003import java.util.function.BinaryOperator;
004
005public class BinaryExpS<R> extends BinaryExp<R, R, R> {
006
007        BinaryExpS(Exp<R> left, Exp<R> right, BinaryOperator<R> operator,String symbol) {
008                super(left, right, operator, symbol);
009        }
010
011        BinaryExpS(BinaryOperator<R> operator, String symbol) {
012                super(operator, symbol);
013        }
014        
015        @Override
016        public BinaryExpS<R> asBinary() {
017                return this;
018        }
019        
020        @Override
021        public Exp<R> clone() {
022                Exp<R> e1 = this.op1;
023                Exp<R> e2 = this.op2;
024                if(e1!= null) e1 = e1.clone();
025                if(e2!= null) e2 = e2.clone();
026                return Exp.createBinaryS(e1, e2, (BinaryOperator<R>) this.operator, this.symbol);
027        }
028        
029}
