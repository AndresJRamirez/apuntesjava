001package us.lsi.common;
002
003import java.util.List;
004
005import com.google.common.base.Function;
006import com.google.common.base.Preconditions;
007import com.google.common.base.Predicate;
008
009public class Tupla3<T1, T2, T3> {
010
011        public static <T1, T2, T3> Tupla3<T1, T2, T3> create(T1 p1, T2 p2, T3 p3) {
012                return new Tupla3<T1, T2, T3>(p1, p2, p3);
013        }
014        
015        public static <T1, T2, T3> Tupla3<T1, T2, T3> create(Tupla3<T1, T2, T3> t) {
016                return new Tupla3<T1, T2, T3>(t.getP1(), t.getP2(), t.getP3());
017        }
018        
019        public static <T> Tupla3<T, T, T> create(List<T> t) {
020                Preconditions.checkArgument(t.size()==3);
021                return create(t.get(0), t.get(1), t.get(2));
022        }
023        
024        public static <T> Tupla3<T, T, T> create(T[] t) {
025                Preconditions.checkArgument(t.length==3);
026                return create(t[0], t[1], t[2]);
027        }
028        
029        public static <T1, T2, T3> Tupla3<T1, T2, T3> flattenP1(Par<Par<T1,T2>,T3> t) {
030                return create(t.getP1().getP1(),t.getP1().getP2(),t.getP2());
031        }
032        
033        public static <T1, T2, T3> Tupla3<T1, T2, T3> flattenP2(Par<T1, Par<T2,T3>> t) {
034                return create(t.getP1(),t.getP2().getP1(),t.getP2().getP2());
035        }
036        
037        
038        private T1 p1;
039        private T2 p2;
040        private T3 p3;
041        
042        private Tupla3(T1 p1, T2 p2, T3 p3) {
043                super();
044                this.p1 = p1;
045                this.p2 = p2;
046                this.p3 = p3;
047        }
048        public T1 getP1() {
049                return p1;
050        }
051        public T2 getP2() {
052                return p2;
053        }
054        public T3 getP3() {
055                return p3;
056        }
057        public Tupla3<T1,T2,T3> setP1(T1 v){
058                return create(v,getP2(),getP3());
059        }
060        
061        public Tupla3<T1,T2,T3> setP2(T2 v){
062                return create(getP1(),v,getP3());
063        }
064        
065        public Tupla3<T1,T2,T3> setP3(T3 v){
066                return create(getP1(),getP2(),v);
067        }
068
069        public <R> Tupla3<R,T2,T3> transformP1(Function<T1,R> f){
070                return create(f.apply(getP1()),getP2(),getP3());
071        }
072        
073        public <R> Tupla3<T1,R,T3> transformP2(Function<T2,R> f){
074                return create(getP1(),f.apply(getP2()),getP3());
075        }
076        
077        public <R> Tupla3<T1,T2,R> transformP3(Function<T3,R> f){
078                return create(getP1(),getP2(),f.apply(getP3()));
079        }
080        
081        public boolean isP1(Predicate<T1> p){
082                return p.apply(getP1());
083        }
084        
085        public boolean isP2(Predicate<T2> p){
086                return p.apply(getP2());
087        }
088        
089        public boolean isP3(Predicate<T3> p){
090                return p.apply(getP3());
091        }
092        
093        @Override
094        public String toString() {
095                return "(" + p1 + "," + p2 + "," + p3 + ")";
096        }
097        @Override
098        public int hashCode() {
099                final int prime = 31;
100                int result = 1;
101                result = prime * result + ((p1 == null) ? 0 : p1.hashCode());
102                result = prime * result + ((p2 == null) ? 0 : p2.hashCode());
103                result = prime * result + ((p3 == null) ? 0 : p3.hashCode());
104                return result;
105        }
106        @Override
107        public boolean equals(Object obj) {
108                if (this == obj)
109                        return true;
110                if (obj == null)
111                        return false;
112                if (getClass() != obj.getClass())
113                        return false;
114                Tupla3<?, ?, ?> other = (Tupla3<?, ?, ?>) obj;
115                if (p1 == null) {
116                        if (other.p1 != null)
117                                return false;
118                } else if (!p1.equals(other.p1))
119                        return false;
120                if (p2 == null) {
121                        if (other.p2 != null)
122                                return false;
123                } else if (!p2.equals(other.p2))
124                        return false;
125                if (p3 == null) {
126                        if (other.p3 != null)
127                                return false;
128                } else if (!p3.equals(other.p3))
129                        return false;
130                return true;
131        }
132        
133        
134}
