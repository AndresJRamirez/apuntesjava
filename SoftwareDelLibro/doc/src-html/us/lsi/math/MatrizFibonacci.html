001package us.lsi.math;
002import java.math.*;
003
004public class MatrizFibonacci {
005        private BigInteger a;
006        private BigInteger b;
007        private static MatrizFibonacci base = 
008                    new MatrizFibonacci(BigInteger.ONE,BigInteger.ZERO);
009        
010        private static BigInteger dos = new BigInteger("2");
011
012        public MatrizFibonacci(BigInteger a1, BigInteger b1){
013                a = a1;
014                b = b1;
015        }
016                
017        public BigInteger getA(){
018                return a;
019        }
020        
021        public BigInteger getB(){
022                return b;
023        }
024        
025        public BigInteger get(int i, int j){
026                BigInteger r;
027                if(i == 0 && j == 0) r = a;
028                else if(i == 1 && j == 0) r = b;
029                else if(i == 0 && j == 1) r = a.add(b);
030                else if(i == 1 && j == 1) r = a;
031                else throw new IllegalArgumentException("Indices no permitidos");
032                return r;
033        }
034        
035        public String toString(){
036                BigInteger amasb = a.add(b); 
037                String s = "(("+amasb+", "+a+"),"+"("+a+", "+b+"))";
038                return s;
039        }
040        
041        public MatrizFibonacci getSquare() {
042                BigInteger a1 = getA();
043                BigInteger b1 = getB();
044                BigInteger ac = a1.multiply(a1);
045                BigInteger bc = b1.multiply(b1);
046                BigInteger ar = ac.add(dos.multiply(a1).multiply(b1));
047                BigInteger br = (ac).add(bc);
048                return new MatrizFibonacci(ar,br);
049        }
050
051        public MatrizFibonacci getMultiplyBase() {
052                return new MatrizFibonacci(a.add(b), a);
053        }
054        
055        public MatrizFibonacci getSum(MatrizFibonacci n2) {
056                BigInteger a1 = getA();
057                BigInteger b1 = getB();
058                BigInteger a2 = n2.getA();
059                BigInteger b2 = n2.getB();
060                return new MatrizFibonacci(a1.add(a2),b1.add(b2));
061        }
062
063        public MatrizFibonacci getZero() {
064                return new MatrizFibonacci(BigInteger.ZERO,BigInteger.ZERO);
065        }
066        
067        public static MatrizFibonacci getOne() {
068                return new MatrizFibonacci(BigInteger.ZERO,BigInteger.ONE);
069        }
070        
071        public static MatrizFibonacci getBase(){
072                return base;
073        }
074
075}
