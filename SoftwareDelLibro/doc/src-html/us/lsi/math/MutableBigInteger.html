001package us.lsi.math;
002
003import java.math.BigInteger;
004import org.apache.commons.lang3.mutable.Mutable;
005
006
007public class MutableBigInteger extends Number implements Comparable<MutableBigInteger>, Mutable<Number> {
008
009      private static final long serialVersionUID = 512176391864L;
010
011      private BigInteger value;
012
013      public MutableBigInteger() {            
014          super();
015      }
016
017      public MutableBigInteger(BigInteger value) {
018          super();
019          this.value = value;
020      }
021
022      public MutableBigInteger(Long value) {
023          super();
024          this.value = new BigInteger(value.toString());
025      }
026      
027      public MutableBigInteger(String value) throws NumberFormatException {
028          super();
029          this.value = new BigInteger(value);
030      }
031
032      @Override
033      public BigInteger getValue() {
034          return value;
035      }
036      
037      public void setValue(BigInteger value) {
038          this.value = value;
039      }
040
041      
042      @Override
043      public void setValue(Number value) {
044          this.value = BigInteger.valueOf(value.longValue());
045      }
046      
047      public void increment() {
048          this.value = this.value.add(BigInteger.ONE);
049      }
050
051      public void decrement() {
052          this.value = this.value.subtract(BigInteger.ONE);
053      }
054      
055      public void add(BigInteger operand) {
056          this.value = this.value.add(operand);     
057      }
058      
059      public void add(Number operand) {
060          this.value = value.add(BigInteger.valueOf(operand.longValue()));
061      }
062      
063      public void subtract(BigInteger operand) {
064          this.value = this.value.subtract(operand);
065      }
066      
067      public void subtract(Number operand) {
068          this.value = value.subtract(BigInteger.valueOf(operand.longValue()));
069      }
070      
071      public void multiply(BigInteger operand) {
072          this.value = this.value.multiply(operand);
073      }
074      
075      public void multiply(Number operand) {
076          this.value = value.multiply(BigInteger.valueOf(operand.longValue()));
077      }
078      
079      @Override
080      public int intValue() {
081          return value.intValue();
082      }
083      
084      @Override
085      public long longValue() {
086          return value.longValue();
087      }
088      
089      @Override
090      public float floatValue() {
091          return value.floatValue();
092      }
093      
094      @Override
095      public double doubleValue() {
096              return value.doubleValue();
097      }
098      
099      public Integer toInteger() {
100          return value.intValue();
101      }
102      
103      @Override
104      public boolean equals(Object obj) {
105          if (obj instanceof MutableBigInteger) {
106              return value.equals(obj);
107          }
108          return false;
109      }
110      
111      @Override
112      public int hashCode() {
113          return value.hashCode();
114      }
115      
116      @Override
117      public int compareTo(MutableBigInteger other) {
118          BigInteger anotherVal = other.value;
119          return value.compareTo(anotherVal);
120      }
121      
122      @Override
123      public String toString() {
124          return value.toString();
125      }
126    
127}
