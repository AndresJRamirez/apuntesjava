001package us.lsi.problemasdelistas;
002
003import static org.junit.Assert.*;
004
005import java.util.Collections;
006import java.util.List;
007
008import org.junit.Test;
009
010import us.lsi.math.Math2;
011import us.lsi.problemasdelistas.ProblemasDeListas.SubSecuencia;
012
013import com.google.common.collect.Lists;
014import com.google.common.collect.Ordering;
015
016
017public class Collections2Test {
018
019        @Test
020        public void testBinarySearchList() {
021                List<Double> lista = Math2.getListDoubleAleatoria(50, -20., 20.);
022                Double key = Math2.getDoubleAleatorio(-20.,20.);
023                ProblemasDeListas.sort(lista);
024                int r1 = ProblemasDeListas.binarySearch(lista, key);
025                int r2 = Collections.binarySearch(lista, key);
026                System.out.println(r1+","+r2);
027                assertTrue(r1==r2);
028        }
029        
030        @Test
031        public void testSortList() {
032                List<Double> lista = Math2.getListDoubleAleatoria(50, -20., 20.);
033                Ordering<Double> ord = Ordering.natural();
034                ProblemasDeListas.sort(lista);
035                assertTrue(ord.isOrdered(lista));
036        }
037
038        @Test
039        public void testSortListComparator() {
040                List<Double> lista = Math2.getListDoubleAleatoria(50, -20., 20.);
041                Ordering<Double> ord = Ordering.natural().reverse();
042                ProblemasDeListas.sort(lista,ord);
043                assertTrue(ord.isOrdered(lista));
044        }
045        
046        @Test
047        public void testEscogeKesimo() {
048                List<Double> lista = Math2.getListDoubleAleatoria(50, -20., 20.);
049                Integer k = Math2.getEnteroAleatorio(0,lista.size());
050                Double r = ProblemasDeListas.getKesimo(lista,k);
051                ProblemasDeListas.sort(lista);
052                assertTrue(lista.get(k).equals(r));
053        }
054
055        @Test
056        public void testEscogeKesimoComparator() {
057                List<Double> lista = Math2.getListDoubleAleatoria(50, -20., 20.);
058                Ordering<Double> ord = Ordering.natural().reverse();
059                Integer k = Math2.getEnteroAleatorio(0,lista.size());
060                Double r = ProblemasDeListas.getKesimo(lista,k,ord);
061                ProblemasDeListas.sort(lista,ord);
062                assertTrue(lista.get(k).equals(r));
063        }
064        
065        @Test
066        public void testSubSecuencia() {
067                Double[] r = {1., -2., 11., -4., 13., -5., 2., 3.};
068                List<Double> lista = Lists.newArrayList();
069                for(Double e:r){
070                        lista.add(e);
071                }
072                SubSecuencia s = ProblemasDeListas.getSubSecuenciaMaxima(lista);
073                assertTrue(new SubSecuencia(lista,2,5).equals(s));
074        }
075}
