001package us.lsi.basictypes;
002
003public class OrType<U, R> {
004
005        public static <U, R> OrType<U, R> fromU(U typeU) {
006                return new OrType<U, R>(typeU, null);
007        }
008
009        public static <U, R> OrType<U, R> fromR(R typeR) {
010                return new OrType<U, R>(null, typeR);
011        }
012        
013        private U typeU;
014        private R typeR;
015        
016        private OrType(U typeU, R typeR) {
017                super();
018                this.typeU = typeU;
019                this.typeR = typeR;
020        }
021
022        public U getU() {
023                return typeU;
024        }
025
026        public R getR() {
027                return typeR;
028        }
029        
030        boolean isU(){
031                return this.typeU != null;
032        }
033        
034        boolean isR(){
035                return this.typeR != null;
036        }
037
038        @Override
039        public int hashCode() {
040                final int prime = 31;
041                int result = 1;
042                result = prime * result + ((typeR == null) ? 0 : typeR.hashCode());
043                result = prime * result + ((typeU == null) ? 0 : typeU.hashCode());
044                return result;
045        }
046
047        @Override
048        public boolean equals(Object obj) {
049                if (this == obj)
050                        return true;
051                if (obj == null)
052                        return false;
053                if (!(obj instanceof OrType))
054                        return false;
055                OrType<?,?> other = (OrType<?,?>) obj;
056                if (typeR == null) {
057                        if (other.typeR != null)
058                                return false;
059                } else if (!typeR.equals(other.typeR))
060                        return false;
061                if (typeU == null) {
062                        if (other.typeU != null)
063                                return false;
064                } else if (!typeU.equals(other.typeU))
065                        return false;
066                return true;
067        }
068        
069        
070}
