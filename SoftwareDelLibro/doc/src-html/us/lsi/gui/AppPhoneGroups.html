028        public static void newGroups() {
029                final Function<String, Person> f = (String s) -> {
030                        String[] ps = s.split(",");
031                        Preconditions.checkArgument(ps.length == 5);
032                        return new Person(ps[0], ps[1], ps[2], ps[3], ps[4]);
033                };
034
035                PhoneGroup group1 = new PhoneGroup("Mecanicos");
036                PhoneGroup group2 = new PhoneGroup("Programadores");
037                Stream2.fromFile("mecanicos.txt")
038                                .<Person> map(f)
039                                .forEach((Person x) -> group1.addPerson(x));
040                Stream2.fromFile("programadores.txt")
041                                .<Person> map(f)
042                                .forEach((Person x) -> group1.addPerson(x));
043
044                categories.addGroup(group1);
045                nameOfCategories.add(group1.getName());
046
047                categories.addGroup(group2);
048                nameOfCategories.add(group2.getName());
049                nameOfCategories.add("Empresarios");
050                nameOfCategories.add("Otros");
051        }
052
053        public static PhoneGroups categories = new PhoneGroups();
054        public static List<String> nameOfCategories = Lists.newArrayList();
055
056        public static PhoneGroups getGroups() {
057                return categories;
058        }
