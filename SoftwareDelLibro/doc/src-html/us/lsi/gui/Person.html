001package us.lsi.gui;
002
003public class Person extends AbstractModelObject {
004        private String m_name;
005        private String m_email;
006        private String m_phone;
007        private String m_mobilePhone1;
008        private String m_mobilePhone2;
009
010        public Person() {
011                m_name = "???????";
012        }
013
014        public Person(String name, String email, String phone, String phone1, String phone2) {
015                m_name = name;
016                m_email = email;
017                m_phone = phone;
018                m_mobilePhone1 = phone1;
019                m_mobilePhone2 = phone2;
020        }
021
022        public String getName() {
023                return m_name;
024        }
025
026        public void setName(String name) {
027                String oldValue = m_name;
028                String oldValueString = toString();
029                m_name = name;
030                firePropertyChange("name", oldValue, m_name);
031                firePropertyChange("string", oldValueString, toString());
032        }
033
034        public String getEmail() {
035                return m_email;
036        }
037
038        public void setEmail(String email) {
039                String oldValue = m_email;
040                String oldValueString = toString();
041                m_email = email;
042                firePropertyChange("email", oldValue, m_email);
043                firePropertyChange("string", oldValueString, toString());
044        }
045
046        public String getPhone() {
047                return m_phone;
048        }
049
050        public void setPhone(String phone) {
051                String oldValue = m_phone;
052                String oldValueString = toString();
053                m_phone = phone;
054                firePropertyChange("phone", oldValue, m_phone);
055                firePropertyChange("string", oldValueString, toString());
056        }
057
058        public String getMobilePhone1() {
059                return m_mobilePhone1;
060        }
061
062        public void setMobilePhone1(String phone1) {
063                String oldValue = m_mobilePhone1;
064                String oldValueString = toString();
065                m_mobilePhone1 = phone1;
066                firePropertyChange("mobilePhone1", oldValue, m_mobilePhone1);
067                firePropertyChange("string", oldValueString, toString());
068        }
069
070        public String getMobilePhone2() {
071                return m_mobilePhone2;
072        }
073
074        public void setMobilePhone2(String phone2) {
075                String oldValue = m_mobilePhone2;
076                String oldValueString = toString();
077                m_mobilePhone2 = phone2;
078                firePropertyChange("mobilePhone2", oldValue, m_mobilePhone2);
079                firePropertyChange("string", oldValueString, toString());
080        }
081
082        public String toString(){
083                return m_name+","+m_email+","+m_phone+","+m_mobilePhone1+","+m_mobilePhone2;
084        }
085        
086        public String getString(){
087                return toString();
088        }
089}
