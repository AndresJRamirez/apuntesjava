001package us.lsi.gui;
002
003import java.awt.Color;
004import java.awt.GridBagConstraints;
005import java.awt.GridBagLayout;
006import java.awt.Insets;
007import java.util.List;
008
009import javax.swing.JButton;
010import javax.swing.JComboBox;
011import javax.swing.JDialog;
012import javax.swing.JLabel;
013import javax.swing.JPanel;
014import javax.swing.border.MatteBorder;
015
016import org.jdesktop.swingbinding.JComboBoxBinding;
017import org.jdesktop.swingbinding.SwingBindings;
018import org.jdesktop.beansbinding.AutoBinding;
019import org.jdesktop.beansbinding.BeanProperty;
020import org.jdesktop.beansbinding.Bindings;
021
022import java.awt.event.ActionListener;
023import java.awt.event.ActionEvent;
024
025
026/**
027 * @author lobas_av
028 * 
029 */
030public class JPhoneGroupDialog extends JDialog {
031        private static final long serialVersionUID = 1L;
032        private final PhoneGroup m_phoneGroup;
033        private final List<String> m_names;
034        private JPanel m_contentPane;
035        private JButton m_buttonOk;
036        @SuppressWarnings("rawtypes")
037        private JComboBox m_comboBox;
038
039        @SuppressWarnings("rawtypes")
040        public JPhoneGroupDialog(List<String> names, PhoneGroup phoneGroup) {
041                m_names = names;
042                m_phoneGroup = phoneGroup;
043                setDefaultCloseOperation(JDialog.DISPOSE_ON_CLOSE);
044                setModal(true);
045                setTitle("Phone Group");
046                setBounds(100, 100, 432, 126);
047                m_contentPane = new JPanel();
048                m_contentPane.setBorder(new MatteBorder(5, 5, 5, 5, (Color) null));
049                setContentPane(m_contentPane);
050                //
051                GridBagLayout gridBagLayout = new GridBagLayout();
052                gridBagLayout.columnWidths = new int[] { 0, 0, 0 };
053                gridBagLayout.rowHeights = new int[] { 0, 0, 0, 0 };
054                gridBagLayout.columnWeights = new double[] { 0.0, 1.0, 1.0E-4 };
055                gridBagLayout.rowWeights = new double[] { 0.0, 0.0, 0.0, 1.0E-4 };
056                m_contentPane.setLayout(gridBagLayout);
057                //
058                {
059                        JLabel lblName = new JLabel("Name:");
060                        GridBagConstraints gbc = new GridBagConstraints();
061                        gbc.anchor = GridBagConstraints.EAST;
062                        gbc.insets = new Insets(5, 5, 5, 5);
063                        gbc.gridx = 0;
064                        gbc.gridy = 0;
065                        m_contentPane.add(lblName, gbc);
066                }
067                {
068                        m_comboBox = new JComboBox();
069                        GridBagConstraints gbc = new GridBagConstraints();
070                        gbc.insets = new Insets(0, 0, 5, 0);
071                        gbc.fill = GridBagConstraints.BOTH;
072                        gbc.gridx = 1;
073                        gbc.gridy = 0;
074                        m_contentPane.add(m_comboBox, gbc);
075                }
076                {
077                        m_buttonOk = new JButton("OK");
078                        m_buttonOk.addActionListener(new ActionListener() {
079                                public void actionPerformed(ActionEvent e) {
080                                        setVisible(false);
081                                }
082                        });
083                        GridBagConstraints gbc = new GridBagConstraints();
084                        gbc.anchor = GridBagConstraints.EAST;
085                        gbc.gridx = 1;
086                        gbc.gridy = 2;
087                        m_contentPane.add(m_buttonOk, gbc);
088                }
089                initDataBindings();
090        }
091        @SuppressWarnings("rawtypes")
092        protected void initDataBindings() {
093                JComboBoxBinding<String, List<String>, JComboBox> jComboBinding = SwingBindings.createJComboBoxBinding(AutoBinding.UpdateStrategy.READ, m_names, m_comboBox);
094                jComboBinding.bind();
095                //
096                BeanProperty<PhoneGroup, String> phoneGroupBeanProperty = BeanProperty.create("name");
097                BeanProperty<JComboBox, String> jComboBoxBeanProperty = BeanProperty.create("selectedItem");
098                AutoBinding<PhoneGroup, String, JComboBox, String> autoBinding = Bindings.createAutoBinding(AutoBinding.UpdateStrategy.READ_WRITE, m_phoneGroup, phoneGroupBeanProperty, m_comboBox, jComboBoxBeanProperty);
099                autoBinding.bind();
100        }
101}
