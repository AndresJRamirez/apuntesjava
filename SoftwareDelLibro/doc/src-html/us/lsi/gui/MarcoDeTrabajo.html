063        public static Double anchura = 800.;
064        public static Double altura = 800.;
065        public static Double xMin;
066        public static Double xMax;
067        public static Double yMin;
068        public static Double yMax;
069        public static Double tamX;
070        public static Double centroX;
071        public static Double tamY;
072        public static Double centroY;
073        public static AffineTransform af = null;
074        
075        public static ObjetoGeometricoAgregado2D objetos = ObjetoGeometricoAgregado2D.create();
076        
077        public static void setMarco(double x1, double x2, double y1, double y2){
078                xMin = x1;
079                xMax = x2;
080                yMin = y1;
081                yMax = y2;
082                tamX = xMax-xMin;
083                centroX = (xMax+xMin)/2;
084                tamY = yMax-yMin;
085                centroY = (yMax+yMin)/2;
086                double ax = anchura/tamX;
087                double ay = - altura/tamY;
088                double bx = - xMin*anchura/tamX;
089            double by = yMax*altura/tamY;
090            System.out.println("Afin ="+ax+","+bx+","+ay+","+by);
091                MarcoDeTrabajo.af = new AffineTransform(ax,0,0,ay,bx,by);
092        }
093        
094        public static void fill(Graphics2D g, Poligono2D p){
095                Polygon polygon = new Polygon();
096                for(Punto2D pt: p.getVertices()){
097                        polygon.addPoint(pt.getX().intValue(), pt.getY().intValue());
098                }
099                g.fillPolygon(polygon);
100        }
101
102        public static void fill(Graphics2D g, Punto2D p){
103                g.fillOval(p.getX().intValue()-4, p.getY().intValue()-4, 8, 8);
104        }
105        
106        public static void draw(Graphics2D g, Punto2D p1, Punto2D p2){
107                g.drawLine(p1.getX().intValue(), p1.getY().intValue(), p2.getX().intValue(), p2.getY().intValue());
108        }
