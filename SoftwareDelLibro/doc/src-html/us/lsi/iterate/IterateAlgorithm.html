001package us.lsi.iterate;
002
003
004import java.util.stream.Stream;;
005
006/**
007 * @author Miguel Toro
008 *
009 * <p> Algoritmo iterativo esen sus diversas formas y sus equivalente recursivo final </p>
010 *
011 * @param <R> Tipo de los datos devueltos por el algoritmo
012 * @param <E> Tipo del estado del algoritmo
013 */
014public class IterateAlgorithm<R,E extends IterateState<R,E>> {
015
016        public static <R, E extends IterateState<R,E>> R iterateStream (E e) {          
017                return Stream.iterate(e, s->s.next())
018                                .filter(s->!s.domain())
019                                .findFirst()
020                                .get()
021                                .ret();                         
022        }
023        
024        public static <R, E extends IterateState<R,E>> R iterateWhile (E e){
025                E st = e;
026                while(st.domain()){
027                        st = st.next();
028                }
029                return st.ret();
030        }
031        
032        public static <R, E extends IterateState<R,E>> R recursiveFinal (E e){
033                R r;
034                if(!e.domain()){
035                        r = e.ret();
036                }else {
037                        r = recursiveFinal(e.next());
038                }
039                return r;
040        }
041
042        
043        
044        public static <R, E extends RecursiveState<R,E>> R recursiveNoFinal (E e){
045                R r;
046                if(!e.domain()){
047                        r = e.ret();
048                }else {
049                        r = e.combine(recursiveNoFinal(e.next()));
050                }
051                return r;
052        }
053        
054        
055}
