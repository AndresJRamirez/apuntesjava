001package us.lsi.iterate;
002
003/**
004 * @author Miguel Toro
005 * 
006 * <p> Tipo que representa la información esencial de un algoritmo iterativo o recursivo final </p>
007 *
008 * @param <R> Tipo del resultado del algoritmo iterativo
009 * @param <E> Tipo que agrega las variables del estado del algoritmo
010 */
011public interface IterateState<R,E extends IterateState<R,E>> {  
012        /**
013         * @return Siguiente estado
014         */
015        E next();
016        /**
017         * @return Valor devuelto por el algoritmo
018         */
019        R ret();
020        /**
021         * @return Verdadero mientras esté dentro del dominio
022         */
023        boolean domain();
024}
