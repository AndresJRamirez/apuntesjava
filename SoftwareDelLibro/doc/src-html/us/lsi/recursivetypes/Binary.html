001package us.lsi.recursivetypes;
002
003public class Binary<T> extends BinaryTree<T> {
004
005        private T label;
006        private BinaryTree<T> left;
007        private BinaryTree<T> right;
008        
009        Binary(T label, BinaryTree<T> left, BinaryTree<T> right) {
010                super();
011                this.label = label;
012                this.left = left;
013                this.right = right;
014        }
015        public T getLabel() {
016                return label;
017        }
018        public void setLabel(T label) {
019                this.label = label;
020        }
021        public BinaryTree<T> getLeft() {
022                return left;
023        }
024        public void setLeft(BinaryTree<T> left) {
025                this.left = left;
026        }
027        public BinaryTree<T> getRight() {
028                return right;
029        }
030        public void setRight(BinaryTree<T> right) {
031                this.right = right;
032        }
033        @Override
034        public String toString() {
035                return "("+label+left+right+")";
036        }
037        
038        public int size(){
039                return this.left.size()+this.right.size()+1;
040        }
041}
