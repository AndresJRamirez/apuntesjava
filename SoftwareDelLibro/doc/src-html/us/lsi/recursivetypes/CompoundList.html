001package us.lsi.recursivetypes;
002
003public class CompoundList<T> extends RList<T> {
004        private T label;
005        private RList<T> list;
006        public CompoundList(T label, RList<T> list) {
007                super();
008                this.label = label;
009                this.list = list;
010        }
011        public T getLabel() {
012                return label;
013        }
014        public void setLabel(T label) {
015                this.label = label;
016        }
017        public RList<T> getList() {
018                return list;
019        }
020        public void setList(RList<T> list) {
021                this.list = list;
022        }       
023}
