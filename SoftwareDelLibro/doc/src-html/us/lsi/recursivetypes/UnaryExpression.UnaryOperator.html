001package us.lsi.recursivetypes;
002
003import java.util.HashMap;
004import java.util.Map;
005
006public class UnaryExpression extends Expression {
007        public enum UnaryOperator{MAS,MENOS}
008        private static Map<UnaryOperator,java.util.function.UnaryOperator<Integer>> map = iniMap();
009        
010        private static Map<UnaryOperator,java.util.function.UnaryOperator<Integer>> iniMap(){
011                Map<UnaryOperator,java.util.function.UnaryOperator<Integer>> map;
012                map = new HashMap<UnaryOperator, java.util.function.UnaryOperator<Integer>>();
013                map.put(UnaryOperator.MAS, (Integer x)->x);
014                map.put(UnaryOperator.MENOS, (Integer x)->-x);
015                return map;
016        }
017        private UnaryOperator operator;
018        private Expression expression;
019        UnaryExpression(UnaryOperator operator, Expression expression) {
020                super();
021                this.operator = operator;
022                this.expression = expression;
023        }
024        public UnaryOperator getOperator() {
025                return operator;
026        }
027        public void setOperator(UnaryOperator operator) {
028                this.operator = operator;
029        }
030        public Expression getExpression() {
031                return expression;
032        }
033        public void setExpression(Expression expression) {
034                this.expression = expression;
035        }
036        
037        @Override
038        public Integer value() {
039                return map.get(this.operator).apply(this.getExpression().value());
040        }       
041        
042        
043}
