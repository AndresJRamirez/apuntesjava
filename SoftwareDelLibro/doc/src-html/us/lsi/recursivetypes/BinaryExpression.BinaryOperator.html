001package us.lsi.recursivetypes;
002
003import java.util.HashMap;
004import java.util.Map;
005
006public class BinaryExpression extends Expression {
007        public enum BinaryOperator{MAS,MENOS,POR,DIV}
008        private static Map<BinaryOperator,java.util.function.BinaryOperator<Integer>> map = iniMap();
009                        
010        private static Map<BinaryOperator,java.util.function.BinaryOperator<Integer>> iniMap(){
011                Map<BinaryOperator,java.util.function.BinaryOperator<Integer>> map;
012                map = new HashMap<BinaryOperator, java.util.function.BinaryOperator<Integer>>();
013                map.put(BinaryOperator.MAS, (Integer x, Integer y)->x+y);
014                map.put(BinaryOperator.MENOS, (Integer x, Integer y)->x-y);
015                map.put(BinaryOperator.POR, (Integer x, Integer y)->x*y);
016                map.put(BinaryOperator.DIV, (Integer x, Integer y)->x/y);
017                return map;
018        }
019        private BinaryOperator operator;
020        private Expression left;
021        private Expression right;
022        BinaryExpression(BinaryOperator operator, Expression left,
023                        Expression right) {
024                super();
025                this.operator = operator;
026                this.left = left;
027                this.right = right;
028        }
029        public BinaryOperator getOperator() {
030                return operator;
031        }
032        public void setOperator(BinaryOperator operator) {
033                this.operator = operator;
034        }
035        public Expression getLeft() {
036                return left;
037        }
038        public void setLeft(Expression left) {
039                this.left = left;
040        }
041        public Expression getRight() {
042                return right;
043        }
044        public void setRight(Expression right) {
045                this.right = right;
046        }
047        
048        @Override
049        public Integer value() {
050                return map.get(this.operator).apply(this.getLeft().value(),this.getRight().value());
051        }       
052}
