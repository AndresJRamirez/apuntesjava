001package us.lsi.algoritmos;
002
003
004import   org.apache.commons.math3.genetics.*;
005
006/**
007 * <p> Algoritmo específico para algoritmos genéticos con codificación real </p>
008 * Si se quieren resolver problemas que se representan por permutacionanes 
009 * los cromosomas que implementemos deben heredar de la clase 
010 * <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/RandomKey.html" target="_blank"> RandomKey </a>
011 * 
012 * 
013 * @author Miguel Toro
014 *
015 * @param <S> Tipo de la solución
016 * @param <C> Tipo del cromosoma
017 * @param <T> Tipo de los objetos a permutar
018 */
019public class AlgoritmoAGRandomKey<S, C extends RandomKey<T>, T> extends
020                AlgoritmoAG<S,Double,C> {
021
022        public AlgoritmoAGRandomKey(ProblemaAG<S, Double, C> problema) {
023                super(problema);
024                super.mutationPolicy = new RandomKeyMutation();
025        }
026
027        protected RandomKey<T> getCromosome() {
028                return super.problema.getChromosome(RandomKey.randomPermutation(DIMENSION));
029        }
030        
031}
