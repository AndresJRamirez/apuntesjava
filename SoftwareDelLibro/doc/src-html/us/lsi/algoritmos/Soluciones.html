001package us.lsi.algoritmos;
002
003import java.util.*;
004import com.google.common.collect.*;
005
006
007/**
008 * Un agregado de elementos ordenados
009 *  
010 * @author Miguel Toro 
011 *
012 * @param <S> El tipo de los elementos
013 */
014public class Soluciones<S> {
015        
016        
017        private SortedSet<S> soluciones;
018
019    private Soluciones(Comparator<S> cmp){
020        soluciones = Sets.<S>newTreeSet(cmp);
021    }
022
023    public static <S extends Comparable<? super S>> Soluciones<S> create() {
024                return new Soluciones<S>(Comparator.<S>naturalOrder());
025        }
026    
027    public static <S> Soluciones<S> create(Comparator<S> cmp) {
028                return new Soluciones<S>(cmp);
029        }
030    
031    public boolean isEmpty(){
032        return soluciones.isEmpty();
033    }
034    
035    public int size(){
036        return soluciones.size();
037    }
038    
039    public boolean add(S s){
040        boolean r = soluciones.add(s);
041        return r;
042    }
043
044        public S getMejorSolucion() {
045                S r = null;
046                if(!soluciones.isEmpty()){
047                        r =soluciones.first();
048                }
049                return r;
050        }
051        
052        public SortedSet<S> getSoluciones() {
053                return soluciones;
054        }
055        
056        public String toString(){
057                String r = "El número de Soluciones es = "+soluciones.size()+".  Las Soluciones son: \n";
058                for(S s : soluciones){
059                        r=r+s+"\n";
060                }
061                return r;
062        }
063}
