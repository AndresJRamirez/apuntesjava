001package us.lsi.algoritmos;
002
003import   org.apache.commons.math3.genetics.*;
004
005/**
006 * <p> Algoritmo específico para algoritmos genéticos con codificación binaria </p>
007 * Los cromosomas que implementemos deben heredar de la clase 
008 * <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/BinaryChromosome.html" target="_blank"> BinaryChromosome </a>
009 * 
010 * 
011 * @author Miguel Toro
012 *
013 * @param <S> Tipo de la solución
014 * @param <C> Tipo del cromosoma
015 */
016public class AlgoritmoAGBinary<S, C extends BinaryChromosome> extends AlgoritmoAG<S,Integer,C> {
017        
018
019        public AlgoritmoAGBinary(ProblemaAG<S, Integer, C> problema) {
020                super(problema);
021                super.mutationPolicy = new BinaryMutation();            
022        }
023
024        protected BinaryChromosome getCromosome() {
025                return super.problema.getChromosome(BinaryChromosome.randomBinaryRepresentation(DIMENSION));
026        }
027        
028}
