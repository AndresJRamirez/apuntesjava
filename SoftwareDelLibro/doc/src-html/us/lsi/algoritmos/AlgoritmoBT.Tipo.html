020/**
021 * <p> Algoritmo que implementa la técnica de Bactracking con sus variantes. 
022 * Un problema que se quiera resolver con esta técnica debe implementar el interface ProblemaBT &lt; S,A &gt;</p>
023 * 
024 * <p>La documentación puede encontarse en el: <a href="../../../document/Tema15.pdf" target="_blank">Tema15</a></p>
025 * 
026 * @author Miguel Toro
027 *
028 * @param <S> El tipo de la solución
029 * @param <A> El tipo de la alternativa
030 * @param <T> El tipo de la propiedad objetivo
031 */
032public class AlgoritmoBT<S, A, T extends Comparable<? super T>> extends AbstractAlgoritmo {
033
034        /**
035         * Max: Maximizar, Min: Minimizar,Otro: No se pretende optimizar
036         *
037         */
038        public static enum Tipo{Max,Min,Otro};
039        
040        /**
041         * El tipo del algoritmo según se pretenda optimizar o no
042         */
043        public static Tipo tipo;
044        /**
045         * Solución obtenida si es única
046         */
047    public S solucion = null;
048    /**
049         * Conjunto de soluciones encontradas
050         */
051        public Set<S> soluciones; 
052        /**
053         * Número de soluciones que se buscan
054         */
055        public static int numeroDeSoluciones = 1;
056        /**
057         * Si se quiere aplicar la técnica aleatoria para escoger una de las alternativas
058         */
059        public static boolean isRandomize = false;
060        /**
061         * Tamaño umbral a partir del cual se escoge aleatoriamente una de las alternativas
062         */
063        public static Integer sizeRef = 10;
064        /**
065         * Si se quiere aplicar la técnica con filtro. En ese caso el Problema debe implementar adicionalmente el interface EstadoBTF
066         */
067        public static  boolean conFiltro = false;
068        /**
069         * Si solo se busca una solución
070         */
071        public static boolean soloLaPrimeraSolucion = true;
072        
073        private ProblemaBT<S,A> problema; 
074    private EstadoBT<S,A> estado;
075    private EstadoBTF<S,A,T> estadoF;
076        private boolean exito = false;
077        private T mejorValor;
078        
079        /**
080         * @param p - El problema a resolver
081         */
082        public AlgoritmoBT(ProblemaBT<S,A> p){
083        problema = p; 
084        mejorValor = null;
085    }
086    
087    /**
088     * Ejecución del algoritmo
089     */
090    @SuppressWarnings("unchecked")
091        public void ejecuta() {
092        soluciones =  Sets.newHashSet();        
093                do {
094                        estado = problema.getEstadoInicial();
095                        if(conFiltro) estadoF = (EstadoBTF<S,A,T>) estado;
096                        bt();
097                } while(isRandomize && soluciones.size()<AlgoritmoBT.numeroDeSoluciones);
098        }
099
100    private Iterable<A> filtraRandomize(EstadoBT<S,A> p, Iterable<A> alternativas){
101        Iterable<A> alt;
102                if(isRandomize && p.size()>sizeRef){
103                        List<A> ls = Lists.newArrayList(alternativas);
104                        List<A> r = Lists.newArrayList();
105                        if(!ls.isEmpty()){
106                                int e = Math2.getEnteroAleatorio(0, ls.size());
107                                r.add(ls.get(e));       
108                        }
109                        alt = r;
110                }else{
111                        alt = alternativas;
112                }
113                return alt;
114        }
115    
116        private void actualizaSoluciones() {
117                if (conFiltro) {
118                        T objetivo = estadoF.getObjetivo();
119                        if (mejorValor == null
120                                        || (AlgoritmoBT.tipo == Tipo.Max
121                                                        && objetivo.compareTo(mejorValor) > 0 || AlgoritmoBT.tipo == Tipo.Min
122                                                        && objetivo.compareTo(mejorValor) < 0)) {
123                                solucion = estado.getSolucion();
124                                soluciones.add(solucion);
125                                mejorValor = objetivo;
126                        }
127                } else {
128                        solucion = estado.getSolucion();
129                        soluciones.add(solucion);
130                }
131        }
132    
133    private void bt() { 
134        if(estado.isFinal()){                   
135                actualizaSoluciones();
136                if(AlgoritmoBT.soloLaPrimeraSolucion  && solucion!=null) exito = true;
137                if(!AlgoritmoBT.soloLaPrimeraSolucion  && soluciones.size()>=AlgoritmoBT.numeroDeSoluciones) exito = true;
138        } else {
139                        for(A a: filtraRandomize(estado,estado.getAlternativas())){                                     
140                                        if(conFiltro && !pasaFiltro(a)) { continue;}
141                                        estado.avanza(a); 
142                                        bt();  
143                                        estado.retrocede(a); 
144                                        if (exito) break;
145                        }
146        }
147   }
148
149    private boolean pasaFiltro(A a){
150        boolean r = true;
151                if(conFiltro){
152                        r= mejorValor==null || 
153                                        AlgoritmoBT.tipo==Tipo.Max && estadoF.getObjetivoEstimado(a).compareTo(mejorValor) > 0 ||
154                                        AlgoritmoBT.tipo==Tipo.Min && estadoF.getObjetivoEstimado(a).compareTo(mejorValor) < 0;
155                }       
156                return r;       
157    }
158    
159        /**
160         * @param p - Predicado con las propiedades de las soluciones buscadas
161         * @return Conjunto de soluciones con las propiedades requeridas
162         */
163        public Set<S> getSolucionesFiltradas(Predicate<S> p) {
164                return soluciones.stream()
165                                .filter(p)
166                                .collect(Collectors.<S>toSet());
167        }

169}
