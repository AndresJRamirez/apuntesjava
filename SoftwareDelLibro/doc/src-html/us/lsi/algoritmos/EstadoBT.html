004/**
005 * <p> Tipo mutable del estado de un problema que se quiere resolver por la técnica de Bactracking
006 * Cada estado tiene asociado un problema del conjunto de problemas </p>
007 * 
008 * <p>La documentación puede encontarse en el: <a href="../../../document/Tema15.pdf" target="_blank">Tema15</a></p>
009 *
010 * @author Miguel Toro
011 *
012 * @param <S> El tipo de la solución
013 * @param <A> El tipo de la alternativa
014 */
015public interface EstadoBT<S, A> {
016        /**
017         * @param a - alternativa escogida para cambiar el estado
018         */
019        void avanza(A a);
020        /**
021         * @param a - alternativa a eliminar para restaurar el estado anterior
022         */
023        void retrocede(A a);
024        /**
025         * @return Tamaño del problema al que está asociado el estado
026         */
027        int size();
028        /**
029         * @return Si el problema asociado al estado es final
030         */
031        boolean isFinal();
032        /**
033         * @constraint Si el estado es Final el iterable  está vacío
034         * @constraint Si el estado no es Final y el iterable está vacío el problema asociado al estado no tiene solución
035         * @return Iterable formado por las alternativas admisibles para el problema asociado
036         */
037        Iterable<A> getAlternativas();
038        /**
039         * @pre isFinal()
040         * @return Solución del problema inicial
041         * @throws IllegalStateException - Si se invoca el método en un estado no final
042         */
043        S getSolucion();
044}
