131         * 
132         *
133         *
134         * @param p Problema
135         * @return AlgoritmoAGReal

137        
138        public static AlgoritmoAGReal createAGReal(ProblemaAGReal<?> p) {
139                RealBinaryChromosome.limites = p.getLimites();
140                RealBinaryChromosome.numeroDeVariables = p.getNumeroDeVariables();
141                RealBinaryChromosome.fitnessFunction = p::fitnessFunction;
142                return new AlgoritmoAGReal(p);
143        }

214        /**
215         * @param fichero Fichero que  describe laq red de flujo
216         * @return Algoritmo de red de flujo que resuelve el problema especificado en el fichero
217         */
218        public static FlowAlgorithm createFlow(String fichero) {
219                FlowGraph f = new FlowGraph(FlowEdge.factory);
220                f = (FlowGraph) GraphsReader.newGraph(fichero, FlowVertex.factory, FlowEdge.factory,f);
221                return FlowAlgorithm.create(f);
222        }
