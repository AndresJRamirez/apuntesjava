001package us.lsi.algoritmos;
002import java.util.*;
003
004import us.lsi.algoritmos.AlgoritmoPD.Sp;
005
006
007/**
008 * <p>Interface que debe implementar un problema para ser resuelto por la técnica de Programación Dinámica </p>
009 * 
010 * <p>La documentación puede encontarse en el: <a href="../../../document/Tema14.pdf" target="_blank">Tema14</a></p>
011 * 
012 * @author Miguel Toro
013 *
014 * @param <S> El tipo de la solución del problema
015 * @param <A> El tipo de las alternativas
016 * @param <T> El tipo del objetivo
017 * @see AlgoritmoPD.Sp
018 */
019public interface ProblemaPD<S,A,T extends Comparable<? super T>>  {     
020        /**
021         * @constraint El tamaño de un problema debe ser mayor que el de cada uno de sus subproblemas
022         * @return El tamaño de un problema
023         */
024        int size();
025        /**
026         * @return Si el problema es un caso base
027         */
028        boolean esCasoBase();
029        /**
030         * @pos Si el problema no tiene solución se debe devolver null
031         * @return La solución del caso base
032         */
033        Sp<A, T> getSolucionCasoBase();
034        /**
035         * @pre ls!=null
036         * @pos Los valores null en ls deben ser eliminados de ls previamente a elegir una alterantiva. Si ls es vacía tras eliminar los valores null
037         * la solución devuelta es null
038         * @param ls - Soluciones de los subproblemas alcanzados tras tomar las diferentes alternativas
039         * @return La solución del problema
040         */
041        Sp<A, T> seleccionaAlternativa(List<Sp<A, T>> ls);
042        /**
043         * @pre 0 &lt; = i &lt; getNumeroSubProblemas(a)
044         * @param a - Alternativa escogida
045         * @param i - Número del subproblema
046         * @return Subproblema
047         */
048        ProblemaPD<S,A,T> getSubProblema(A a, int i);
049        /**
050         * @pre ls.size() == getNumeroSubProblemas(a) y a está incluida en getAlternativas()
051         * @pos  Si un elemento de ls es null la solución devuelta es también null
052         * @param a - Alternativa escogida
053         * @param ls - Soluciones de los subproblemas
054         * @return La solución para un problema si las soluciones para los subproblemas están en ls 
055         * y se escoge la alternativa a
056         */
057        Sp<A, T> combinaSolucionesParciales(A a , List<Sp<A, T>> ls);   
058        /**
059         * @pos Si un problema no tiene solución el conjunto de alternativas es vacío 
060         * @return Las alternativas disponibles para el problema
061         */
062        Iterable<A> getAlternativas();
063        /**
064         * @param a - Alternativa escogida
065         * @return Número de subproblemas para una alternativa dada
066         */
067        int getNumeroSubProblemas(A a);
068        /**
069         * @pre El problema es un caso base
070         * @param sp - Solución parcial del caso base
071         * @return La solución para un caso base si la solución parcial es a. Si no tiene solución se devuelve null. 
072         */
073        S getSolucionReconstruida(Sp<A, T> sp); 
074        /**
075         * @pre El problema no es un caso base, ls !=null y ls.size() &lt; 0 y a  una de las alternativas del problema
076         * @param sp - Solución parcial del problema
077         * @param ls - Soluciones de los subproblemas
078         * @return La solución para un problema si las soluciones para los subproblemas están en ls 
079         * y la solución parcial del problema es  a. Si un problema no tiene solución se devuelve null. Un problema 
080         * que tiene algún subproblema sin solución no tiene solución tampoco
081         */
082        S getSolucionReconstruida(Sp<A, T> sp, List<S> ls);     
083}
