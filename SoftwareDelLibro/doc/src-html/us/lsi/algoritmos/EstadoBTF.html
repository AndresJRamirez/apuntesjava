003/**
004 * <p> Tipo mutable del estado de un problema que se quiere resolver por la técnica de Bactracking con Filtro </p>
005 * 
006 * <p> La documentación puede encontarse en el: <a href="../../../document/Tema15.pdf" target="_blank">Tema15</a></p>
007 * 
008 * @author Miguel Toro
009 *
010 * @param <S> - Tipo de la solución
011 * @param <A> - Tipo de las alternativas
012 * @param <T> - Tipo del objetivo
013 */
014public interface EstadoBTF<S, A, T extends Comparable<? super T>> extends EstadoBT<S, A> {
015        /**
016         * @pre isFinal()
017         * @return - Valor de la propiedad objetivo del problema inicial
018         * @throws IllegalStateException - Si se invoca el método en un estado no final
019         */
020        T getObjetivo();
021        /**
022         * 
023         * @param a - Alternativa escogida
024         * @return - Valor estimado para la propiedad objetivo del problema inicial si en el problema actual se toma la alternativa a.
025         *           
026         */
027        T getObjetivoEstimado(A a);
028}
