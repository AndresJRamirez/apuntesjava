001package us.lsi.beans;
002
003import java.beans.PropertyChangeEvent;
004import java.beans.PropertyVetoException;
005import java.beans.VetoableChangeListener;
006
007public class ConsolaConVeto implements VetoableChangeListener {
008        
009        private static int nc = 1;
010        private int id;
011        
012        ConsolaConVeto() {
013                super();
014                // TODO Auto-generated constructor stub
015                id = nc;
016                nc++;
017        }
018
019        @Override
020        public void vetoableChange(PropertyChangeEvent  evt) throws PropertyVetoException {
021                // TODO Auto-generated method stub
022        
023                String property = evt.getPropertyName();
024                System.out.println(this+",  "+ property+", "+evt.getNewValue()+"  en  "+evt.getSource());
025                if(property.equals("x")){
026                        Double x = (Double) evt.getNewValue();
027                        if(x < 0.){
028                                System.out.println("Veto sobre "+property+", "+evt.getNewValue()+"  en  "+evt.getSource());
029                                throw new PropertyVetoException(property, evt);
030                        } 
031                }else if(property.equals("y")){
032                        Double y = (Double) evt.getNewValue();
033                        if(y < 100.) {
034                                System.out.println("Veto sobre "+property+", "+evt.getNewValue()+"  en  "+evt.getSource());
035                                throw new PropertyVetoException(property, evt);
036                        } 
037                }else if(property.equals("distanciaAlOrigen")){
038                        Double d = (Double) evt.getNewValue();
039                        if(d < 10.) {
040                                System.out.println("Veto sobre "+property+", "+evt.getNewValue()+"  en  "+evt.getSource());
041                                throw new PropertyVetoException(property, evt);
042                        } 
043                }
044                
045        }
046
047        @Override
048        public String toString() {
049                return "ConsolaConVeto [id=" + id + "]";
050        }
051
052}
