262                autoBinding_3.bind();
263                //
264                BeanProperty<ListenablePunto, Double> listenablePuntoBeanProperty = BeanProperty.create("x");
265                BeanProperty<JTextField, String> jTextFieldBeanProperty = BeanProperty.create("text");
266                AutoBinding<ListenablePunto, Double, JTextField, String> autoBinding = Bindings.createAutoBinding(UpdateStrategy.READ_WRITE, p1, listenablePuntoBeanProperty, xText, jTextFieldBeanProperty);
