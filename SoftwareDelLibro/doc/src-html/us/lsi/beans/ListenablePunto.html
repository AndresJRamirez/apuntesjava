001package us.lsi.beans;
002
003import java.beans.PropertyChangeListener;
004import java.beans.PropertyChangeSupport;
005
006
007import us.lsi.geometria.Punto2D;
008
009public class ListenablePunto extends Punto2D {
010        
011        final private PropertyChangeSupport boundProperty = new PropertyChangeSupport(this);
012        private static int n = 1;
013        public int id;
014        
015        ListenablePunto() {
016                super();
017                id = n;
018                n++;
019        }
020        
021        ListenablePunto(Double x, Double y) {
022                super(x, y);
023                id = n;
024                n++;
025        }
026        
027        public void addPropertyChangeListener(String property,PropertyChangeListener listener) {
028                boundProperty.addPropertyChangeListener(property, listener);
029        }
030        
031        public void removePropertyChangeListener(String property,PropertyChangeListener listener) {
032                boundProperty.removePropertyChangeListener(property, listener);
033        }
034
035        
036        public void addPropertyChangeListener(PropertyChangeListener listener) {
037                boundProperty.addPropertyChangeListener(listener);
038        }
039
040        public void removePropertyChangeListener(PropertyChangeListener listener) {
041                boundProperty.removePropertyChangeListener(listener);
042        }
043
044        @Override
045        public void setX(Double x) {
046                Double oldX = getX();
047                Double oldD = super.getDistanciaAlOrigen();
048                super.setX(x);
049                boundProperty.firePropertyChange("x", oldX, x);
050                boundProperty.firePropertyChange("distanciaAlOrigen", oldD, super.getDistanciaAlOrigen());
051        }
052        
053        @Override
054        public void setY(Double y) {
055                // TODO Auto-generated method stub
056                Double oldY = getY();
057                Double oldD = super.getDistanciaAlOrigen();
058                super.setY(y);
059                boundProperty.firePropertyChange("y", oldY, y);
060                boundProperty.firePropertyChange("distanciaAlOrigen", oldD, super.getDistanciaAlOrigen());
061        }
062        
063
064        @Override
065        public String toString() {
066                return "ListenablePunto [id = "+id+", getX()=" + getX() + ", getY()=" + getY() + "]";
067        }
068
069        
070}
