001package us.lsi.beans;
002
003import us.lsi.math.Math2;
004
005public class ListenableRunnablePunto extends ListenablePunto implements Runnable {
006
007        private boolean fin = false;
008        private Thread thread = new Thread(this);
009        
010        public ListenableRunnablePunto() {
011                // TODO Auto-generated constructor stub
012        }
013
014        public ListenableRunnablePunto(Double x, Double y) {
015                super(x, y);
016                // TODO Auto-generated constructor stub
017        }
018        
019        @Override
020        public void run() {
021                // TODO Auto-generated method stub
022                double r = 0.;
023                while(!fin){
024                        setX(r+1);
025                        try {
026                                Thread.sleep(Math2.escogeEntre(0., 2000.));
027                        } catch (InterruptedException e) {
028                                // TODO Auto-generated catch block
029                                e.printStackTrace();
030                        }
031                        setY(r*r);
032                        r=r+2;
033                }               
034                
035        }
036
037        public boolean isFin() {
038                return fin;
039        }
040
041        public void setFin(boolean fin) {
042                this.fin = fin;
043        }
044
045        public void start() {
046                thread.start();
047        }
048
049        public final void stop() {
050                fin = true;
051        }
052        
053
054}
