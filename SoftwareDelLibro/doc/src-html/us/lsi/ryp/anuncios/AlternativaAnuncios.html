001package us.lsi.ryp.anuncios;
002
003
004public class AlternativaAnuncios {
005        
006        public enum Opcion {Insertar, Eliminar, Intercambiar}
007        
008
009        public Opcion opcion;
010        public Integer p1;
011        public Integer p2;
012        
013        public static AlternativaAnuncios createInsertar(Integer p1,Integer p2) {
014                return new AlternativaAnuncios(Opcion.Insertar, p1, p2);
015        }
016        
017        public static AlternativaAnuncios createEliminar(Integer p1) {
018                return new AlternativaAnuncios(Opcion.Eliminar, p1, null);
019        }
020        
021        public static AlternativaAnuncios createIntercambiar(Integer p1, Integer p2) {
022                return new AlternativaAnuncios(Opcion.Intercambiar, p1, p2);
023        }
024        
025        private AlternativaAnuncios(Opcion opcion, Integer p1, Integer p2) {
026                super();
027                this.opcion = opcion;
028                this.p1 = p1;
029                this.p2 = p2;
030        }
031
032        @Override
033        public String toString() {
034                String s;
035                if(this.opcion.equals(Opcion.Insertar) || this.opcion.equals(Opcion.Intercambiar)){
036                        s = "AlternativaAnuncios [opcion=" + this.opcion + "," + this.p1 + ","+ this.p2 +"]";
037                } else {
038                        s = "AlternativaAnuncios [opcion=" + this.opcion + "," + this.p1+"]";
039                }
040                return s;
041        }
042}
